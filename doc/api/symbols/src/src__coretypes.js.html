  9     /**
 10      * Creates a new Enum.
 11      * 
 12      * @example
 13      * 
 14      * var WeekDays = new tent.coreTypes.Enum('FRIDAY,SATURDAY,SUNDAY');
 15      * assertTrue(WeekDays.FRIDAY === 1 &&
 16 	 * 	 WeekDays.SATURDAY === 2 &&
 17      * 	 WeekDays.SUNDAY == 3);
 18      * 
 19      * // flag enum
 20      * var FontStyles = new tent.coreTypes.Enum('_FLAGS,BOLD,ITALIC,UNDERLINE,SMALLCAPS');
 21      * assertTrue(FontStyles.BOLD == 1 &&
 22      *   FontStyles.ITALIC == 2 &&
 23      *   FontStyles.UNDERLINE == 4 &&
 24      *   FontStyles.SMALLCAPS == 8);
 25      * 
 26      * // get names
 27      * assertEquals('SATURDAY',WeekDays.getName(2));
 28      * assertEquals('ITALIC|SMALLCAPS',FontStyles.getName(10));
 29      * 
 30      * @class an Enum of Number values
 31      * @constructor
 32      * @param arguments Enum names, string arguments, comma separated allowed (eg: ('ONE,TWO,THREE','FOUR')), use '_FLAGS' special name to create a flag based enum (every name gets a Math.pow(2,n) value)
 33      */
 34     tent.coreTypes.Enum = function Enum(){
 35         this.__names__ = [];
 36         this.__values__ = {};
 37 		/**
 38 		 * Indicates if flag enabled values are used (ie Math.pow(2,n))
 39 		 * @field
 40 		 */
 41         this.__flags__ = false;
 42         this.__enum = true; // identify as Enum, e.g. provides intellisense is VisualStudio
 43         if (arguments.length > 0) {
 44             this.add.apply(this, arguments);
 45         }
 46     }
 47     
 48     /**
 49      * @param {Number} value
 50      * @return {String} the name of a numeric value (if it's a combination of flags, returns a pipe separated string)
 51      */
 52     tent.coreTypes.Enum.prototype.getName = function(value){
 53     
 54         if (this.__names__) {
 55             if (this.__names__[value]) {
 56                 return this.__names__[value];
 57             }
 58             else 
 59                 if (this.__flags__) {
 60                     var v = value, n = '';
 61                     for (var name in this.__values__) {
 62                         if (this.__values__[name] && ((this.__values__[name] & v) === this.__values__[name])) {
 63                             v -= this.__values__[name];
 64                             if (n) {
 65                                 n += '|';
 66                             }
 67                             n += name;
 68                         }
 69                     }
 70                     if (n && (v === 0)) {
 71                         return n;
 72                     }
 73                     else 
 74                         if (n) {
 75                             return n + '|' + v;
 76                         }
 77                 }
 78         }
 79     };
 80 	
 81     tent.coreTypes.Enum.prototype.__freeValue__ = function(){
 82         var free = (typeof this.__maxValue__ == 'undefined' ? 1 : this.__maxValue__ + 1);
 83         if (this.__values__ && typeof this.__values__[free] != 'undefined') {
 84             // __maxValue is out-dated, recalculate
 85             for (var n in this.__values__) {
 86                 if (this.__values__[n] >= free) {
 87                     this.__maxValue__ = this.__values__[n];
 88                     free = this.__values__[n] + 1;
 89                 }
 90             }
 91         }
 92         if (this.__flags__) {
 93             if (free < 1) {
 94                 free = 1;
 95             }
 96             else {
 97                 // return the next valid flag (Math.pow(2,x))
 98                 free = Math.pow(2, Math.ceil(Math.log(free) / Math.log(2)));
 99             }
100         }
101         return free;
102     };

105      * Activates/Deactivates the use of flag enabled values (Math.pow(2,n))
106      * @param {Boolean} use flag enabled values?
107      * @return {tent.Enum} this Enum
108      */
109     tent.coreTypes.Enum.prototype.useFlags = function(uf){
110         this.__flags__ = uf;
111         return this;
112     };

114     var __setNameValue__ = function(_enum, name, value){
115         // warning: only for internal use, doesnt validate for conflicts!
116         _enum[name] = value;
117         _enum.__values__[name] = value;
118         if (typeof _enum.__names__[value] == 'undefined') {
119             _enum.__names__[value] = name;
120         }
121         if ((typeof _enum.__maxValue__ == 'undefined') || _enum.__maxValue__ < value) {
122             _enum.__maxValue__ = value;
123         }
124     };

126     /**
127      * Adds names to this Enum
128      * @param arguments Enum names, string arguments, comma separated allowed (eg: ('ONE,TWO,THREE','FOUR')), use '_FLAGS' special name to create a flag based enum (every name gets a Math.pow(2,n) value)
129      * @return {tent.Enum} this Enum
130      */
131     tent.coreTypes.Enum.prototype.add = function(){
132     
133         if (arguments.length < 1) {
134             return this;
135         }
136         
137         var names, name;
138         
139         for (var argi = 0, argl = arguments.length; argi < argl; argi++) {
140         
141             var data = arguments[argi];
142             
143             if (data instanceof Array) {
144                 // add from names array
145                 this.add.apply(this, data);
146             }
147             else 
148                 if (typeof data == 'object') {
149                     // add from values dict
150                     for (name in data) {
151                         if (data.hasOwnProperty(name)) {
152                             var v = this.__values__[name];
153                             if ((!v instanceof Number) || (typeof this.__names__[v] != 'undefined')) {
154                                 v = this.__freeValue__();
155                             }
156                             __setNameValue__(this, name, v);
157                         }
158                     }
159                 }
160             if (typeof data == 'string') {
161                 // add names, allow comma-separated names
162                 if (data) {
163                     names = data.split(',');
164                     for (var i = 0, l = names.length; i < l; i++) {
165                         name = names[i].replace(/^\s+/g, '').replace(/\s+$/g, ''); // trim
166                         if (name == '_FLAGS') {
167                             // special name to activate flags behavior
168                             this.useFlags(true);
169                         }
170                         else {
171                             if (typeof this.__values__[name] != 'undefined') {
172                                 if (this.__values__[name] instanceof Number) {
173                                     throw 'Enum duplicate name found: ' + name;
174                                 }
175                                 else {
176                                     throw 'Enum invalid name (reserved): ' + name;
177                                 }
178                             }
179                             if (i == l - 1 && argi < argl - 1 && typeof arguments[argi + 1] == 'number') {
180                                 // add Enum name=specified value
181                                 __setNameValue__(this, name, arguments[argi + 1]);
182                                 argi++;
183                             }
184                             else {
185                                 // add new Enum name=value
186                                 __setNameValue__(this, name, this.__freeValue__());
187                             }
188                         }
189                     }
190                 }
191             }
192         }
193         return this;
194     };

196     var NameCounter_NodeToString = function(){
197         var s = this._name + ((this._count > 1) ? '(' + this._count + ')' : '');
198         if (this._childrenCount) {
199             s += '{';
200             var i = 0;
201             for (var n in this) {
202                 if (n.substr(0, 1) != '_' && typeof this[n] == 'object') {
203                     if (i > 0) {
204                         s += ', ';
205                     }
206                     s += this[n];
207                     i++;
208                 }
209             }
210             s += '}';
211         }
212         return s;
213     };

216     /**
217      * Creates a new NameCounter
218      * 
219      * @example
220      * 
221      * var counter = new tent.coreTypes.NameCounter();
222      * counter.add('reptiles.frog');
223      * counter.add('mammals.cat',4);
224      * counter.add('reptiles.frog');
225      * counter.add('mammals.rat',8);
226      * counter.add('reptiles',2);
227      * 
228      * // check totals
229      * assertEquals('animals=(15){mammals(12){cat(4), rat(8)}, reptiles(4){frog(2)}}',
230      * 		'animals='+counter);
231      * 
232      * @class a NameCounter that holds a (hierarchical) collection of named numeric counters, use {@link #toString} to get the current human-readable totals
233      * @constructor
234      * @param {String} name Counter root Name
235      */
236     tent.coreTypes.NameCounter = function NameCounter(name){
237         this.root = {
238             _name: name || '',
239             _count: 0,
240             toString: NameCounter_NodeToString
241         };
242     };
243     
244     tent.coreTypes.NameCounter.prototype.__nodeAdd__ = function(node, inc){
245     
246         node._count = (node._count || 0) + inc;
247         if (node._parent) {
248             if (node._count == 0 && !node._childrenCount) {
249                 if (node._parent._childrenCount) {
250                     node._parent._childrenCount--;
251                 }
252                 delete node._parent[node._name];
253             }
254             this.__nodeAdd__(node._parent, inc);
255         }
256     };
257     
258     /**
259      * Increments the count for a certain name
260      * @param {String} name the name of the counter, a hierarchical counter can be specified using dot-separated name
261      * @param {Number} inc
262      */
263     tent.coreTypes.NameCounter.prototype.add = function(name, inc){
264         if (typeof inc != "number") {
265             inc = 1;
266         }
267         
268         var namePath = name.split('.');
269         var node = this.root;
270         for (var i = 0; i < namePath.length; i++) {
271             if (typeof node[namePath[i]] == 'undefined') {
272                 node._childrenCount = (node._childrenCount || 0) + 1;
273                 node[namePath[i]] = {
274                     _parent: node,
275                     _name: namePath[i],
276                     _count: 0,
277                     toString: NameCounter_NodeToString
278                 };
279             }
280             node = node[namePath[i]];
281         }
282         
283         this.__nodeAdd__(node, inc);
284         
285     };
286     
287     /**
288      * Resets all counters
289      */
290     tent.coreTypes.NameCounter.prototype.reset = function(){
291         this.root = {
292             _name: this.root._name,
293             _count: 0,
294             toString: NameCounter_NodeToString
295         };
296     };
297     
298     tent.coreTypes.NameCounter.prototype.toString = function(){
299         return NameCounter_NodeToString.apply(this.root);
300     };
