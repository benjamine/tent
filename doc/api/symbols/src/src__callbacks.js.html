1 /**
  2  * Callback utilities
  3  * @namespace Callback utilities
  4  * @requires tent
  5  * @requires tent.arrays
  6  * @requires tent.events
  7  * @name tent.callback
  8  */
  9 tent.declare('tent.callback', function(){
 10 
 11     /**
 12      * Creates a new Callback Handler
 13      * @param {Object} [options] callback options
 14      * @param {function()} [options.complete] a function to call on complete with result as argument
 15      */
 16     tent.callback.start = function(options){
 17         return new tent.callback.CallbackHandler(options);
 18     }
 19     
 20     /**
 21      * Creates a new Callback Handler
 22      * @class handles the execution and finalization of function callbacks
 23      * @param {Object} [options] callback options
 24      * @param {function()} [options.complete] a function to call on complete with result as argument
 25      */
 26     tent.callback.CallbackHandler = function CallbackHandler(options){
 27         this.options = options || {};
 28         this.result = {};
 29         this.context = {};
 30         this.completed = false;
 31         tent.events.makeBindable(this);
 32         this.bind('complete', function(){
 33             if (typeof this.options.complete == 'function') {
 34                 this.options.complete(this.result);
 35             }
 36         });
 37     }
 38     
 39     /**
 40      * Adds an error to the results of this callback
 41      * @param {Object} error
 42      */
 43     tent.callback.CallbackHandler.prototype.addError = function(error){
 44         if (this.result.errors) {
 45             this.result.errors = [];
 46         }
 47         this.result.errors.push(error);
 48         return this;
 49     }
 50     
 51     /**
 52      * Completes the execution of this callback and reports results
 53      */
 54     tent.callback.CallbackHandler.prototype.complete = function(){
 55         if (!this.completed) {
 56             if (!this.result.errors || this.result.errors.length < 1) {
 57                 this.result.ok = true;
 58             }
 59             this.trigger('beforeComplete');
 60             this.completed = true;
 61             this.trigger('complete');
 62             this.trigger('afterComplete');
 63         }
 64         return this;
 65     }
 66 });
