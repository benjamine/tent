1 /**
  2  * @requires tent.connectors.rest
  3  * @name tent.connectors.couchdb
  4  * @namespace Apache CouchDB RESTful Persister
  5  */
  6 tent.declare('tent.connectors.couchdb', function(){
  7 
  8     /**
  9      * Creates a new CouchDB RESTful connector
 10      * @class a CouchDB RESTful connector
 11      * @extends tent.connectors.rest.RestConnector
 12      * @constructor
 13      */
 14     tent.connectors.couchdb.CouchDBConnector = function CouchDBConnector(){
 15     
 16         var ctr = new tent.connectors.rest.RestConnector();
 17         
 18         ctr.couchdb = true;
 19         ctr.provider = 'CouchDB';
 20         
 21         ctr.options.saveChanges.bulk = true;
 22         ctr.options.saveChanges.bulkUrl = '_bulk_docs';
 23         ctr.options.saveChanges.useCollectionInUrl = false;
 24         ctr.options.saveChanges.revisionProperty = '_rev';
 25         
 26         ctr.filters.load.unshift(function(d){
 27             if (typeof d.data.rows == 'object' && d.data.rows instanceof Array &&
 28             typeof d.data.total_rows == 'number') {
 29                 // document set
 30                 d.data.items = [];
 31                 for (var i = 0, l = d.data.rows.length; i < l; i++) {
 32                     d.data.items.push(d.data.rows[i].doc);
 33                 }
 34                 delete d.data.rows;
 35             }
 36             else {
 37                 // single document
 38                 d.data = {
 39                     total_rows: 1,
 40                     items: [d.data]
 41                 };
 42             }
 43             return d;
 44         });
 45         
 46         ctr.filters.saveChanges.push(function(d){
 47             if (d.data && d.data.items instanceof Array) {
 48                 for (var i = 0, l = d.data.items.length; i < l; i++) {
 49                     var doc = d.data.items[i];
 50                     if (d.packedChanges && d.packedChanges[i] &&
 51                     d.packedChanges[i].__changeState__ === tent.entities.ChangeStates.DELETED) {
 52                         doc._deleted = true;
 53                     }
 54                 }
 55                 d.data.docs = d.data.items;
 56                 delete d.data.items;
 57             }
 58             else 
 59                 if (d.packedChanges && d.packedChanges.length === 1) {
 60                     if (d.options.http.type === 'DELETE') {
 61                         if (!d.options.http.headers) {
 62                             d.options.http.headers = {};
 63                         }
 64                         d.packedChanges[0]._deleted = true;
 65                         d.options.http.headers["If-Match"] = d.packedChanges[0][d.options.saveChanges.revisionProperty];
 66                         d.options.http.url += "?rev=" + d.packedChanges[0][d.options.saveChanges.revisionProperty];
 67                     }
 68                 }
 69             return d;
 70         });
 71         
 72         ctr.filters.saveResult.unshift(function(d){
 73             for (var i = 0, l = d.data.items.length; i < l; i++) {
 74                 var doc = d.data.items[i];
 75                 
 76                 delete doc.ok;
 77                 
 78                 doc._id = doc.id;
 79                 delete doc.id;
 80                 doc._rev = doc.rev;
 81                 delete doc.rev;
 82                 
 83                 if (!doc.error && d.packedChanges && d.packedChanges[i] &&
 84                 d.packedChanges[i].__changeState__ === tent.entities.ChangeStates.DELETED) {
 85                     doc._deleted = true;
 86                 }
 87             }
 88             return d;
 89         });
 90         
 91         ctr.filters.saveResult.unshift(function(d){
 92             if (typeof d.data == 'object' && d.data instanceof Array) {
 93                 // document set
 94                 d.data = {
 95                     items: d.data
 96                 };
 97             }
 98             else {
 99                 // single document
100                 d.data = {
101                     items: [d.data]
102                 };
103             }
104             return d;
105         });
106         
107         return ctr;
108     }
109 });
