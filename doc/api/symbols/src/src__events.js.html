1 /**
  2  * Event emitting
  3  * @namespace Event emitting
  4  * @requires tent
  5  * @requires tent.arrays
  6  * @name tent.events
  7  */
  8 tent.declare('tent.events', function(){
  9 
 10 
 11     /**
 12      * Extends all argument objects as event emitters. 
 13      * Adds all functions in {@link tent.events.extendFunctions}
 14      * @example
 15      * 
 16      * 		var dog = { name: 'Bobby' }, cat = {name: 'Kitty'}, status;
 17      * 		tent.events.extend(dog);
 18      * 		dog.bind({bark: function(e, at) {
 19      * 			status = this.name+ ' barked at '+at.name;
 20      * 		});
 21      * 		dog.trigger('bark',cat);
 22      * 
 23      * 		equals(status, "Bobby barked at Kitty", "bark event was handled");
 24      * 
 25      */
 26     tent.events.extend = function(){
 27         for (var i = 0; i < arguments.length; i++) {
 28             var arg = arguments[i];
 29 	        if (typeof arg == 'object' && !arg.__events__) {
 30 				// add private event map
 31 	            arg.__events__ = {};
 32 				// add functions
 33 	            for (fname in tent.events._extendFunctions){
 34 					arg[fname]= tent.events.extendFunctions[fname];
 35 				}
 36 	        }
 37         }
 38         return arguments[0];
 39     }
 40 		
 41     /**
 42      * Unextends event emitters.
 43      */
 44     tent.events.unextend = function(){
 45         for (var i = 0; i < arguments.length; i++) {
 46             var arg = arguments[i];
 47 	        if (typeof arg == 'object' && arg.__events__) {
 48 	            delete arg.__events__;
 49 	            for (fname in tent.events._extendFunctions){
 50 					delete arg[fname];
 51 				}
 52 	        }
 53         }
 54     }
 55     
 56 	/**
 57 	 * Functions added to event emitters (see {@link tent.events.extend})
 58 	 * @name tent.events.extendFunctions
 59 	 * @namespace Functions added to event emitters (see {@link tent.events.extend})
 60 	 */
 61     tent.events.extendFunctions = {    
 62     
 63 		/**
 64 		 * Binds functions to an event (or events)
 65 		 * 
 66 		 * @example
 67 		 * 
 68 		 * 		// bind by name
 69 		 * 		obj.bind('click', function(e) {
 70 		 * 			alert(this.id+' clicked at '+e.data.position);
 71 		 * 		};
 72 		 * 
 73 		 * 		// bind with a map
 74 		 * 		obj.bind({
 75 		 * 			activate: function(e) {
 76 		 *				console.log('activated');
 77 		 * 		}, desactivate: function(e) {
 78 		 * 				console.log('deactivated');
 79 		 * 		});
 80 		 * 
 81 		 * 		// bind passing data
 82 		 * 		obj.bind('load', { context: this.title }, function(e) {
 83 		 * 			alert(e.data.context+' clicked at '+e.data.position);
 84 		 * 		};
 85 		 * 
 86 		 * @param {String} eventType
 87 		 * @param {Object} [data] data to include in eventData, if an order field is included it's used to sort handler invocations
 88 		 * @param {Object|function()} handler
 89 		 * @return the context object (this) for chaining
 90 		 */
 91         bind: function(eventType, data, handler){
 92         
 93             if (typeof eventType == 'object') {
 94                 // first param is a hash of events and handlers
 95                 for (var et in eventType) {
 96                     if (eventType.hasOwnProperty(et)) {
 97                         this.bind(et, eventType[et]);
 98                     }
 99                 }
100                 return this;
101             }
102             
103             if (typeof handler == 'undefined') {
104                 handler = data;
105                 data = null;
106             }
107             
108             if (!handler) {
109                 throw 'a handler function or object is required';
110             }
111             if ((typeof handler != 'function') && (typeof handler.handle != 'function')) {
112                 throw 'handler must be function(data) or a handler object (with a "handle(sender,data)" function)';
113             }
114             
115             var eventTypes = eventType.split(' ');
116             var order = data ? (data.order || 0) : 0;
117             for (var eti = 0; eti < eventTypes.length; eti++) {
118                 var evType = eventTypes[eti];
119                 if (evType) {
120                     if (!this.__events__[evType]) {
121                         this.__events__[evType] = {
122                             handlers: []
123                         };
124                     }
125                     var handlers = this.__events__[evType].handlers;
126                     // find position to insert handler based on order
127                     var position = handlers.length;
128                     for (var hi = handlers.length - 1; hi >= 0; hi--) {
129                         var h = handlers[hi];
130                         if ((h.order || 0) <= order) {
131                             position = hi + 1;
132                             break;
133                         }
134                     };
135                     
136                     handlers.splice(position, 0, {
137                         order: order,
138                         data: data,
139                         handler: handler
140                     });
141                 }
142             }
143             return this;
144         },
145         
146         /**
147          * Triggers an event notifying suscribers, aditional arguments are passed to handlers.
148          * Propagation can be stopped calling by handlers using: e.stopImmediatePropagation(); 
149          * @param {String} eventType an event type name
150 		 * @return the context object (this) for chaining
151          */
152         trigger: function(eventType){
153             if (!this.__events__[eventType]) {
154                 return;
155             }
156             var handlers = this.__events__[eventType].handlers;
157 
158             var e = {
159                 target: this,
160                 type: eventType
161             };
162 
163 			/**
164 			 * @inner
165 			 */
166 			var stopProp = function(){
167                 this.immediatePropagationStopped = true;
168             };
169 			e.stopImmediatePropagation = stopProp;
170 			
171             var hargs = Array.prototype.slice.call(arguments, 1);
172             hargs.unshift(e);
173             
174             for (var hi = 0, hl = handlers.length; hi < hl; hi++) {
175                 var h = handlers[hi];
176                 e.data = h.data;
177                 if (typeof h.handler == 'function') {
178                     try {
179                         e.result = h.handler.apply(this, hargs);
180                     } 
181                     catch (err) {
182                         console.error(err);
183                     }
184                 }
185                 else 
186                     if (typeof h.handler.handle == 'function') {
187                         try {
188                             e.result = h.handler.handle.apply(h.handler, hargs);
189                         } 
190                         catch (err) {
191                             console.error(err);
192                         }
193                     }
194                 
195                 if (e.immediatePropagationStopped) {
196                     break;
197                 }
198             }
199             return this;
200         },
201         
202 		/**
203 		 * Removes an event binding.
204 		 * If only event type is specified all binding to that type are remove.
205 		 * If only a handler is specified the handler is removed from all event types
206 		 * If both are specified the handler is removed from the specific event type
207 		 * @param {String} [eventType] event type name to unbind
208 		 * @param {Object|function()} [handler] a handler object or function to unbind
209 		 * @return the context object (this) for chaining
210 		 */
211         unbind: function(eventType, handler){
212         
213             if (typeof handler == 'undefined' && typeof eventType == 'function') {
214                 handler = eventType;
215                 eventType = null;
216             }
217             
218             if (typeof handler == 'undefined') {
219                 if (typeof eventType == 'undefined') {
220                     this.__events__ = {};
221                 }
222                 else {
223                     delete this.__events__[eventType];
224                 }
225             }
226             else {
227                 if (eventType) {
228                     if (!this.__events__[eventType]) {
229                         return;
230                     }
231                     var handlers = this.__events__[eventType].handlers;
232                     
233                     for (var hi = handlers.length; hi >= 0; hi--) {
234                         var h = handlers[hi];
235                         if (h.handler === handler) {
236                             handlers.splice(hi, 1);
237                         }
238                     }
239                 }
240                 else {
241                     for (var et in this.__events__) {
242                         var handlers = this.__events__[et].handlers;
243                         
244                         for (var hi = handlers.length; hi >= 0; hi--) {
245                             var h = handlers[hi];
246                             if (h.handler === handler) {
247                                 handlers.splice(hi, 1);
248                             }
249                         }
250                     }
251                 }
252             }
253             return this;
254         }
255         
256     };
257     
258 });
