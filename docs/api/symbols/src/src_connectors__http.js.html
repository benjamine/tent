1 /**
  2  * @requires tent.connectors
  3  * @name tent.connectors.http
  4  * @namespace HTTP utilities
  5  */
  6 tent.declare('tent.connectors.http', function(){
  7 
  8     /**
  9      * Combines URIs
 10      * @return {String} the URI that combines all arguments
 11      */
 12     tent.connectors.http.uriCombine = function(){
 13         var uri = '';
 14         for (var i = 0, l = arguments.length; i < l; i++) {
 15             var strarg;
 16 			if (typeof arguments[i] == 'undefined' || arguments[i]=== null){
 17 				strarg = '';
 18 			} else {
 19 				strarg = arguments[i] + '';
 20 			}
 21             if (strarg) {
 22             
 23                 if (strarg.match(/^[A-Za-z]+\:\/\//)) {
 24                     // is absolute, replace current uri
 25                     uri = strarg;
 26                 }
 27                 else {
 28                     // strip base uri query and hash
 29                     var stripIndex = uri.indexOf('?');
 30                     var hashIndex = uri.indexOf('#');
 31                     if ((hashIndex >= 0) && (stripIndex < 0 || stripIndex > hashIndex)) {
 32                         stripIndex = hashIndex;
 33                     }
 34                     if (stripIndex >= 0) {
 35                         uri = uri.substr(0, stripIndex);
 36                     }
 37                     
 38                     if (uri.substr(uri.length - 1, 1) == '/') {
 39                         if (strarg.substr(0, 1) == '/') {
 40                             uri += strarg.substr(1);
 41                         }
 42                         else 
 43                             if (strarg.substr(0, 2) == './') {
 44                                 uri += strarg.substr(2);
 45                             }
 46                             else {
 47                                 uri += strarg;
 48                             }
 49                     }
 50                     else {
 51                         if (strarg.substr(0, 1) == '/') {
 52                             uri += strarg;
 53                         }
 54                         else 
 55                             if (strarg.substr(0, 2) == './') {
 56                                 uri += strarg.substr(1);
 57                             }
 58                             else {
 59                                 uri += '/' + strarg;
 60                             }
 61                     }
 62                 }
 63             }
 64         }
 65         return uri;
 66     };
 67     
 68     /**
 69      * Default options for http requests
 70      */
 71     tent.connectors.http.requestDefaultOptions = {
 72         type: 'GET',
 73         cache: false
 74     };
 75     
 76     /**
 77      * Makes an HTTP request using node.js libraries
 78      * @param {Object} options
 79      * @param {function()} callback
 80      */
 81     tent.connectors.http.request_nodejs = function(options, callback){
 82         throw 'http request using node.js not implemented'
 83     };
 84     
 85     /**
 86      * Makes an HTTP request using jQuery.ajax
 87      * @param {Object} options
 88      * @param {function()} callback
 89      */
 90     tent.connectors.http.request_jquery = function(options, callback){
 91     
 92         if (typeof jQuery == 'undefined') {
 93             throw 'jQuery is required for ajax requests';
 94         }
 95         
 96         var opt = tent.combineOptions(tent.connectors.http.requestDefaultOptions, options);
 97         
 98         jQuery.ajax({
 99             context: {
100 				options: opt,
101 				callback: callback
102 			},
103             url: tent.connectors.http.uriCombine(opt.baseUrl, opt.url),
104             beforeSend: function(req){
105                 if (opt.auth) {
106                     if (typeof Base64 == 'undefined') {
107                         throw 'Base64 is required for http basic authentication';
108                     }
109                     var usrpwd64 = Base64.encode(opt.username + ':' + opt.password)
110                     req.setRequestHeader("Origin", document.location.protocol + "//" + document.location.host);
111                     req.setRequestHeader("Authorization", "Basic " + usrpwd64);
112                 }
113             },
114             type: opt.type,
115             cache: !!opt.cache,
116             dataType: 'json',
117 			contentType: 'application/json',
118             data: JSON.stringify(opt.data),
119             headers: opt.headers,
120             timeout: opt.timeout,
121             username: opt.username,
122             password: opt.password,
123             success: function(data, textStatus, req){
124                 if (typeof this.callback == 'function') {
125                     this.callback({
126 						ok: true,
127 						data: data,
128 						req: req
129 					});
130                 }
131             },
132             error: function(req, textStatus, error){
133                 if (typeof this.callback == 'function') {
134                     this.callback({
135 						error: error || 'error',
136 						req: req
137 					});
138                 }
139             }
140         });
141     };
142     
143     /**
144      * Makes an HTTP request
145      * @param {Object} options
146      * @param {function()} callback
147      */
148     tent.connectors.http.request = tent.connectors.http.request_jquery;
149     
150 });
