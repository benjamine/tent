 18 /**
 19  * Creates a Namespace
 20  * @class a Namespace that groups classes, functions, constants and children Namespaces
 21  * @constructor
 22  * @param {Namespace} [parent] parent Namespace
 23  * @param {String} path a namespace path, eg 'web.ajax.json.parsing'
 24  */
 25 tent.Namespace = function Namespace(parent, path){
 26 
 27     if (!path) {
 28         if (!parent) {
 29             throw 'must specify namespace path';
 30         }
 31         else {
 32             path = parent;
 33             parent = tent.global;
 34         }
 35     }
 36     
 37     if (path === '..') {
 38         if (parent === tent.global) {
 39             throw 'cannot use \'..\' at top level';
 40         }
 41         return parent.parent;
 42     }
 43     
 44     var dot1 = path.indexOf('.');
 45     var dot2 = (dot1 < 0 ? -1 : path.indexOf('.', dot1 + 1));
 46     
 47     var fullPath = (parent === tent.global ? '' : parent.fullname + '');
 48     if (fullPath) {
 49         fullPath += '.';
 50     }
 51     fullPath += path;
 52     
 53     if (dot1 < 0) {
 54         if (typeof parent[path] == 'undefined') {
 55             this.fullname = fullPath;
 56             this.name = path;
 57             this.parent = parent;
 58             parent[path] = this;
 59         }
 60         else 
 61             if (!parent[path] instanceof tent.Namespace) {
 62                 throw 'error creating namespace: "' + fullPath + '" is not a Namespace object';
 63             }
 64             else {
 65                 return parent[path];
 66             }
 67     }
 68     else {
 69         return new tent.Namespace(new tent.Namespace(parent, path.slice(0, dot1)), path.slice(dot1 + 1));
 70     }
 71     return this;
 72 };

 90 /**
 91  * Expands this Namespace
 92  * @param {function()|Object} expander a function that receives this Namespace as parameter and adds members (eg: classes, constants, functions) or an Object wich properties are copied into this Namespace
 93  */
 94 tent.Namespace.prototype.expand = function(expander){
 95     if (expander instanceof Function) {
 96         expander(this);
 97     }
 98     else 
 99         if (typeof expander == 'object') {
100             for (var propName in expander) {
101                 if (expander.hasOwnProperty(propName)) {
102                     if (typeof this[propName] != 'undefined') {
103                         throw this.fullname + '.' + propName + ' already exists';
104                     }
105                     this[propName] = expander[propName];
106                 }
107             }
108         }
109     return this;
110 };

112 /**
113  * Declares Namespaces and add members to them  (Namespace paths are unique, ie: the same Namespace object is returned for the same path)
114  *
115  * <p>Parameters can be any sequence of:
116  * 	  <ul>
117  *    <li>{@link tent.Namespace}: changes the current Namespace</li>
118  *    <li>{string} path: changes current Namespace to a subnamespace of the current Namespace (creating Namespaces if needed)</li>
119  *    <li>{function()|Object} expander: a function that adds members to the current Namespace or an Object wich properties are copied to the current Namespace</li>
120  *    </ul>
121  *	</p>
122  * @param {tent.Namespace} [parent] parent Namespace
123  * @param {String}  [path] Namespace path, eg: 'web.ajax.json.parsing'
124  * @param {function()|Object}  [expander] a function that receives the created Namespace and adds members  (eg: classes, constants, functions) or an Object wich properties are copied to the current Namespace
125  * @return {tent.Namespace} the last current Namespace
126  */
127 tent.declare = function(/*[parent], [path], [expander]*/){
128 
129     var ns;
130     
131     for (var i = 0; i < arguments.length; i++) {
132         var a = arguments[i];
133         if (a instanceof tent.Namespace) {
134             ns = a;
135         }
136         else 
137             if (typeof a == 'string') {
138                 if (!ns) {
139                     ns = new tent.Namespace(a);
140                 }
141                 else {
142                     ns = new tent.Namespace(ns, a);
143                 }
144             }
145             else 
146                 if (a instanceof Function || typeof a == 'object') {
147                     if (!ns) {
148                         throw 'no namespace provided';
149                     }
150                     ns.expand(a);
151                 }
152                 else {
153                     throw 'invalid argument type at index ' + i + ', expected Namespace, string, Function';
154                 }
155     }
156     
157     return ns;
158 }

160 /**
161  * Extends a class with another class (mixin)
162  * @param {Object} target class to extend
163  * @param {Object} mixin mixin class to use as source
164  * @return {Object} the extended class
165  */
166 tent.mixin = function(target, mixin){
167 	for (var name in mixin.prototype){
168 		if (mixin.prototype.hasOwnProperty(name)){
169 			target.prototype[name] = mixin.prototype[name];
170 		}
171 	}
172 	return target;
173 }

175 /**
176  * Combines option objects
177  */
178 tent.combineOptions = function(){
179 	var options = {};
180 	for (var i=0, l= arguments.length; i<l; i++ ){
181 		var arg = arguments[i];
182 		for (var name in arg){
183 			if (arg.hasOwnProperty(name)){
184 				options[name] = arg[name];
185 			}
186 		}
187 	}
188 	return options;
189 }

229 /**
230  * Clones objects
231  *
232  * This is useful in browsers that only support property change tracking on DOM objects (eg: IE8)
233  * @param {Object} obj object or array to clone
234  * @param {Object} [options] cloning options
235  * @param {Boolean} [options.onlyOwnProperties] clone only own properties (see Object.hasOwnProperty)
236  * @param {Boolean} [options.dom] clone as DOM objects (eg: span elements)
237  * @param {Boolean} [options.deep] clone deeply, traverse through object properties?
238  * @param {Boolean} [options.clonePrivates] clone private members (properties starting with _)
239  * @param {Boolean} [options.skipPrivates] skip (don't copy) private members (properties starting with _)
240  * @param {Boolean} [options.attachedObjectsIds] if true objects attached to a {@link tent.entities.Context} are replaced by their ids (useful for persistence serialization)
241  * @param {Boolean} [options.attachedObjects] if true objects attached to a {@link tent.entities.Context} are cloned too, by default is true (if false deep cloning stops at attached objects)
242  * @param {Boolean} [options.ignoreCircularReferences] if cloning deeply, ignore circular references (otherwise error is thrown)
243  * @param {Boolean} [options.onlyForTracking] only clone if the current browser only support tracking on DOM objects (eg: IE8), otherwise original obj is returned
244  * @return {Object} a clone of obj (or obj if no cloning was performed)
245  */
246 tent.clone = function(obj, options){
247     if (obj === null) {
248         return null;
249     }
250     else 
251         if (typeof obj == 'undefined') {
252             return undefined;
253         }
254         else 
255             if (typeof obj == 'object') {
256             
257                 if (options && options.onlyForTracking &&
258                 tent.changes.getPropertyInterceptMode() != tent.changes.PropertyInterceptModes.DEFINEPROPERTYONLYDOM) {
259                     return obj;
260                 }
261                 
262                 if (!options) {
263                     options = {};
264                 }
265                 else 
266                     if (options.deep && !options.deepStack) {
267                         options.deepStack = tent.arrays.extend([]);
268                     }
269                 
270                 if (obj instanceof Array) {
271                     var cloneObj = [];
272                     for (var i = 0; i < obj.length; i++) {
273                         if (typeof obj[i]=='object' && options.deep) {
274                             options.deepStack.push(obj);
275                             if (options.deepStack.lastIndexOf(obj[i]) < 0) { 
276 				                if ((options.attachedObjectsIds) && obj[i].__collection__) {
277 									cloneObj.push(tent.entities.getId(obj[i]));
278 								}else if ((options.attachedObjects !== false) || !obj[i].__collection__) {
279 									cloneObj.push(tent.clone(obj[i], options));
280 								}
281                             } else if (!options.ignoreCircularReferences) {
282 								throw 'Circular reference when cloning array';
283 							}
284                             options.deepStack.pop();
285                         }
286                         else {
287                             cloneObj.push(obj[i]);
288                         }
289                     }
290                     return cloneObj;
291                 }
292                 else {
293                     var cloneObj = options.dom ? document.createElement('span') : {};
294                     for (var pname in obj) {
295 						if (options.onlyOwnProperties && !obj.hasOwnProperty(pname)){
296 							continue;
297 						}						
298 						
299 						var isPrivate =  pname.substr(0, 1) == '_';
300                         if (options.skipPrivates && isPrivate) {
301                             continue;
302                         }else if ((!options.clonePrivates) && isPrivate) {
303 							var value = obj.__observable__ ? tent.pget(obj,pname): obj[pname];
304 			                if (typeof value=='object' && options.attachedObjectsIds && value.__collection__) {
305 	                            cloneObj[pname] = tent.entities.getId(value);
306 								continue;
307 							}
308 			                if (typeof value=='object' && (options.attachedObjects === false) && value.__collection__) {
309 								continue;
310 							}
311                             cloneObj[pname] = value;
312                         }
313                         else {
314                         
315 							var value = obj.__observable__ ? tent.pget(obj,pname): obj[pname];
316 						
317                             if (options.deep && typeof value == 'object') {
318 				                if ((options.attachedObjectsIds) && value.__collection__) {
319 									cloneObj[pname] = tent.entities.getId(value);
320 									continue;
321 								}
322 				                if ((options.attachedObjects === false) && value.__collection__) {
323 									continue;
324 								}
325                                 options.deepStack.push(obj);
326                                 if (options.deepStack.indexOf(value) < 0) {
327                                     cloneObj[pname] = tent.clone(value, options);
328                             	} else if (!options.ignoreCircularReferences) {
329 									throw 'Circular reference when cloning object';
330 								}
331                                 options.deepStack.pop();
332                             }
333                             else {
334                                 cloneObj[pname] = value;
335                             }
336                         }
337                         
338                     }
339                     return cloneObj;
340                 }
341             }
342     return obj;
343 }
