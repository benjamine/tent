 84         for (var i = this.listeners.length; i >= 0; i--) {

 92 	/**
 93 	 * Binds a callback function to events of this Log
 94 	 * @param {function()} callback 
 95 	 * @param {Number} [level]
 96 	 * @return {tent.logging.Log} this
 97 	 */
 98     tent.logging.Log.prototype.bind = function(callback, level){
 99         if (!callback instanceof Function) {
100             throw 'a callback Function must be provided';
101         }
102         
103         var lvl = level;
104         if (typeof lvl != 'number') {
105             if (typeof lvl == 'undefined') {
106                 lvl = tent.logging.Levels.TRACE;
107             }
108             else {
109                 lvl = tent.logging.Levels[lvl] || tent.logging.Levels.TRACE;
110             }
111         }
112         
113         var found = false;
114         for (var i = this.listeners.length - 1; i >= 0; i--) {
115             var list = this.listeners[i];
116             if (list.callback === callback) {
117                 // already bound, just update level
118                 list.level = lvl;
119                 found = true;
120             }
121         };
122         if (!found) {
123             this.listeners.push({
124                 callback: callback,
125                 level: lvl
126             })
127         }
128         
129         return this;
130     }

135 	 * @return {tent.logging.Log} this
136 	 */
137     tent.logging.Log.prototype.unbind = function(callback){
138         if (!callback instanceof Function) {
139             throw 'a callback Function must be provided';
140         }
141         for (var i = this.listeners.length; i >= 0; i--) {
142             if (this.listeners[i].callback === callback) {
143                 this.listeners.splice(i, 1);
144             }
145         }
146         return this;
147     }

149 	/**
150 	 * Notifies an event to Log listeners
151 	 * @private
152 	 * @param {String} message
153 	 * @param {Number} level
154 	 */	
155     tent.logging.Log.prototype.notify = function(message, level){
156     
157         var lvl = level;
158         if (typeof lvl != 'number') {
159             if (!lvl) {
160                 lvl = tent.logging.Levels.INFO;
161             }
162             else {
163                 lvl = tent.logging.Levels[lvl] || tent.logging.Levels.INFO;
164             }
165         }
166         
167         for (var i = 0, l = this.listeners.length; i < l; i++) {
168             var lst = this.listeners[i];
169             if (lst.callback) {
170                 if ((typeof lst.level == 'undefined') || (lst.level <= lvl)) {
171                     lst.callback(message, level);
172                 }
173             }
174         }
175     }

225 	/**
226 	 * Stringifies all arguments
227 	 * @private
228 	 * @return {String} a String represeting all arguments
229 	 */
230     tent.logging.Log.prototype.stringify = function(){
231     
232         var s = '';
233         
234         try {
235         
236             var options = (arguments.length > 0 && arguments[arguments.length - 1] &&
237             arguments[arguments.length - 1]._options) ? arguments[arguments.length - 1] : {
238                 _options: true
239             };
240             
241             if (!options.deepStack) {
242                 options.deepStack = [];
243                 tent.arrays.extend(options.deepStack);
244             }
245             
246             for (var i = 0, l = arguments.length; i < l; i++) {
247                 if (arguments[i] && (typeof arguments[i] == 'object') && arguments[i]._options) {
248                     continue;
249                 }
250                 if (arguments[i] instanceof Function) {
251                     continue;
252                 }
253                 if (s) {
254                     s += ', ';
255                 }
256                 if (options.deepStack.length > 3 || options.deepStack.lastIndexOf(arguments[i]) >= 0) {
257                     s += '#';
258                 }
259                 else {
260                     if (typeof arguments[i] == 'undefined') {
261                         s += 'undefined';
262                     }
263                     else 
264                         if (arguments[i] == null) {
265                             s += 'null';
266                         }
267                         else 
268                             if (arguments[i] instanceof Array) {
269                                 var arg = tent.arrays.functions.clone.apply(arguments[i]).filter(function(item){
270                                     if (item instanceof Function) {
271                                         return false;
272                                     }
273                                     if (options.deepStack.lastIndexOf(item >= 0)) {
274                                         return false;
275                                     }
276                                     return true;
277                                 });
278                                 arg.push(options);
279                                 options.deepStack.push(arguments[i]);
280                                 s += '[';
281                                 if (options.deepStack.length < 4) {
282                                     s += this.stringify.apply(this, arg);
283                                 }
284                                 options.deepStack.pop();
285                                 s += ']';
286                             }
287                             else 
288                                 if (typeof(arguments[i]) == 'string') {
289                                     s += '"' + arguments[i] + '"';
290                                 }
291                                 else 
292                                     if (arguments[i] instanceof Date) {
293                                         s += '"' + arguments[i] + '"';
294                                     }
295                                     else 
296                                         if (typeof(arguments[i]) == 'object') {
297                                             var so = '';
298                                             options.deepStack.push(arguments[i]);
299                                             if (options.deepStack.length < 4) {
300                                                 for (var n in arguments[i]) {
301                                                     if (arguments[i][n] instanceof Function) {
302                                                         continue;
303                                                     }
304                                                     if (options.deepStack.lastIndexOf(arguments[i][n]) >= 0) {
305                                                         continue;
306                                                     }
307                                                     if (so) {
308                                                         so += ', ';
309                                                     }
310                                                     so += '"' + n + '": ' + this.stringify.apply(this, [arguments[i][n], options]);
311                                                 }
312                                             }
313                                             options.deepStack.pop();
314                                             s += '{' + so + '}';
315                                         }
316                                         else {
317                                             s += arguments[i];
318                                         }
319                 }
320             }
321             
322         } 
323         catch (err) {
324             return '#err#';
325         }
326         return s;
327     }

329 	/**
330 	 * If a String argument is used returns the same string, otherwise the stringified arguments (as {@link #stringify})
331 	 * @private
332 	 * @return {String} a String represeting all arguments 
333 	 */
334     tent.logging.Log.prototype.stringOrStringify = function(){
335         if (arguments.length == 1 && typeof arguments[0] == 'string') {
336             return arguments[0];
337         }
338         else {
339             return this.stringify.apply(this, arguments);
340         }        
341     }

343 	/**
344 	 * @private
345 	 * @param {Object} message
346 	 * @param {Object} level
347 	 */
348     tent.logging.consoleLogger = function(message, level){
349     
350         if (level == tent.logging.Levels.TRACE) {
351             if (console.trace) {
352                 console.trace(message);
353             }
354             else {
355                 console.info(message);
356             }
357         }
358         if (level == tent.logging.Levels.DEBUG) {
359             if (console.debug) {
360                 console.debug(message);
361             }
362             else {
363                 console.info(message);
364             }
365         }
366         if (level == tent.logging.Levels.INFO) {
367             console.info(message);
368         }
369         if (level == tent.logging.Levels.WARN) {
370             if (console.warn) {
371                 console.warn(message);
372             }
373             else {
374                 console.info(message);
375             }
376         }
377         if (level == tent.logging.Levels.ERROR) {
378             console.error(message);
379         };
380         if (level == tent.logging.Levels.FATAL) {
381             console.error(message);
382         }
383     }

394 	/**
395 	 * @private
396 	 * @param {Object} outputElement
397 	 */
398     tent.logging.createHtmlAppendLogger = function(outputElement){
399         var f=null;
400         if (outputElement.tagName == 'INPUT') {
401 			/**
402 			 * @inner
403 			 */
404             f = function(message, level){
405                 outputElement.value += '\n[' + tent.logging.Levels.getName(level) + '] ' + message;
406             }
407         }
408         else 
409             if (outputElement.tagName == 'UL') {
410 				/**
411 				 * @inner
412 				 */
413                 f = function(message, level){
414                     var li = document.createElement('li');
415                     li.innerHTML = '[' + tent.logging.Levels.getName(level) + '] ' + message;
416                     outputElement.appendChild(li);
417                 }
418             }
419             else 
420                 if (outputElement.tagName == 'TABLE') {
421 					/**
422 					 * @inner
423 					 */
424                     f = function(message, level){
425                         var tr = document.createElement('tr');
426                         var td1 = document.createElement('td');
427                         td1.setAttribute('class', 'eventType');
428                         td1.innerHTML = tent.logging.Levels.getName(level);
429                         tr.appendChild(td1);
430                         var td2 = document.createElement('td');
431                         td2.setAttribute('class', 'eventMessage');
432                         
433                         td2.innerHTML = message.replace(/(\<)/g, '<').replace(/(\>)/g, '>').replace(/(\&)/g, '&').replace(/(\s)/g, ' ');
434                         
435                         tr.appendChild(td2);
436                         
437                         var tbody = outputElement.tBodies[0];
438                         tbody.appendChild(tr);
439                     }
440                 }
441                 else {
442 					/**
443 					 * @inner
444 					 */
445                     f = function(message, level){
446                         outputElement.innerHTML += '<br/>[' + tent.logging.Levels.getName(level) + '] ' + message;
447                     }
448                 }
449         f.outputElement = outputElement;
450         return f;
451     }
