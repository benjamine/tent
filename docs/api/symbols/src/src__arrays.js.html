1 /**
  2  * Array utilities
  3  * @namespace Array utilities
  4  * @requires tent
  5  * @name tent.arrays
  6  * @namespace Array utilities
  7  */
  8 tent.declare('tent.arrays', function(){
  9 
 10     /**
 11      * Extends an Array with all the functions defined on {@link tent.arrays.functions}
 12      * @param {Array} array
 13      * @param {Boolean} [override] override if function already exists? (false by default, native implementations are faster)
 14      * @return {Array} the same array extended
 15      */
 16     tent.arrays.extend = function(array, override){
 17         for (var fname in tent.arrays.functions) {
 18             if (override || !(array[fname] instanceof Function)) {
 19                 array[fname] = tent.arrays.functions[fname];
 20             }
 21         }
 22         return array;
 23     };
 24     
 25     /**
 26      * Common Array functions
 27      * @namespace Common Array functions
 28      */
 29     tent.arrays.functions = {
 30 
 31 		/**
 32 		 * finds an item by a property 
 33 		 * @param {String} propertyName
 34 		 * @param value
 35 		 * @return the first item with the specified value on the specified property, or null if no item is found
 36 		 */
 37         findByProperty: function(propertyName, value){
 38             for (var i = 0, l = this.length; i < l; i++) {
 39                 if (this[i][propertyName] === value) {
 40                     return this[i];
 41                 }
 42             }
 43             return null;
 44         },        
 45     	
 46 		/**
 47 		 * finds an item by a property 
 48 		 * @param {String} propertyName
 49 		 * @param value
 50 		 * @return {Number} the index of the first item with the specified value on the specified property, or -1 if no item is found
 51 		 */
 52         indexByProperty: function(propertyName, value){
 53             for (var i = 0, l = this.length; i < l; i++) {
 54                 if (this[i][propertyName] === value) {
 55                     return i;
 56                 }
 57             }
 58             return -1;
 59         },  
 60 		
 61 		/**
 62 		 * @param {Object} item an item to search for
 63 		 * @return {Number} the first index where item is found, or -1 if item is not found
 64 		 */
 65         indexOf: function(item){
 66             for (var i = 0, l = this.length; i < l; i++) {
 67                 if (this[i] === item) {
 68                     return i;
 69                 }
 70             }
 71             return -1;
 72         },
 73         
 74 		/**
 75 		 * @param {Object} item an item to search for
 76 		 * @return {Number} the last index where item is found, or -1 if item is not found
 77 		 */
 78         lastIndexOf: function(item){
 79             for (var i = this.length - 1; i > 0; i--) {
 80                 if (this[i] === item) {
 81                     return i;
 82                 }
 83             }
 84             return -1;
 85         },
 86         
 87 		/**
 88 		 * Creates a new Array filtering items of this Array
 89 		 * @param {function()} condition a function that returns true for elements to include in results
 90 		 * @return {Array} a new array containing items that satisfied the condition
 91 		 */
 92         filter: function(condition){
 93             var filtered = [];
 94             for (var i = 0, l = this.length; i < l; i++) {
 95                 if (condition(this[i])) {
 96                     filtered.push(this[i]);
 97                 }
 98             }
 99             return filtered;
100         },
101         
102         /**
103          * removes all references to argument items from the array, returns true if one or more items are removed
104          * @return {Boolean} true if any item is removed
105          */
106         remove: function(){
107             var removed;
108             for (var i = 0, l = this.length; i < l; i++) {
109                 for (var ai = 0, al = arguments.length; ai < al; ai++) {
110                     if (this[i] == arguments[ai]) {
111                         this.splice(i, 1);
112                         removed = true;
113                         i--;
114                         l--;
115                         break;
116                     }
117                 }
118             }
119             if (removed) {
120                 return true;
121             }
122 			return false;
123         },
124         
125         /**
126          * equivalent to: this[i] = item
127          * @param {Object} i index to set
128          * @param {Object} item value to set at index
129          */
130         set: function(i, item){
131             this.splice(i, 1, item);
132         },
133         
134         /**
135          * push arguments, filtering already existing items
136          * @return {Number} resulting length of this Array
137          */
138         pushUnique: function(){
139             if (arguments.length == 1) {
140                 // optimized push for 1 argument only
141                 var found = false;
142                 for (var i = 0, l = this.length; i < l; i++) {
143                     if (this[i] == arguments[0]) {
144                         found = true;
145                         break;
146                     }
147                 }
148                 if (!found) {
149                     this.push(arguments[0]);
150                 }
151             }
152             else {
153                 var exists = [];
154                 for (var i = 0, l = this.length; i < l; i++) {
155                     for (var ai = 0, al = arguments.length; ai < al; ai++) {
156                         if (this[i] == arguments[ai]) {
157                             exists[ai] = true;
158                         }
159                     }
160                 }
161                 for (var ai = 0, al = arguments.length; ai < al; ai++) {
162                     if (!exists[ai]) {
163                         this.push(arguments[ai]);
164                     }
165                 }
166             }
167             return this.length;
168         },
169         
170 		/**
171 		 * @return {Array} a new array with the same contents
172 		 */
173         clone: function(){
174             var cl = [];
175             Array.prototype.push.apply(cl, this);
176             return cl;
177         },
178         
179         /**
180          * @param {Object} array an array to compare with
181          * @return {Boolean} true if both arrays contains the same elements
182          */
183         isCloneOf: function(array){
184             if (!array) {
185                 return false;
186             }
187             if (this.length != array.length) {
188                 return false;
189             }
190             if (this === array) {
191                 return false;
192             }
193             
194             for (var i = this.length; i >= 0; i--) {
195                 if (this[i] !== array[i]) {
196                     return false;
197                 }
198             }
199             return true;
200         }
201         
202     };
203     
204 });
