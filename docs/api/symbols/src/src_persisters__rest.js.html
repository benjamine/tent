 10      * Combines URIs (only combines for paths, query and hash combining not supported)
 11      * @return {String} the URI that combines all arguments
 12      */
 13     tent.persisters.rest.uriCombine = function(){
 14         var uri = '';
 15         for (var i = 0, l = arguments.length; i < l; i++) {
 16             var strarg = arguments[i] + '';
 17             if (strarg) {
 18             
 19                 if (strarg.match(/^[A-Za-z]+\:\/\//)) {
 20                     // is absolute, replace current uri
 21                     uri = strarg;
 22                 }
 23                 else {
 24                     if (uri.substr(uri.length - 1, 1) == '/') {
 25                         if (strarg.substr(0, 1) == '/') {
 26                             uri += strarg.substr(1);
 27                         }
 28                         else 
 29                             if (strarg.substr(0, 2) == './') {
 30                                 uri += strarg.substr(2);
 31                             }
 32                             else {
 33                                 uri += strarg;
 34                             }
 35                     }
 36                     else {
 37                         if (strarg.substr(0, 1) == '/') {
 38                             uri += strarg;
 39                         }
 40                         else 
 41                             if (strarg.substr(0, 2) == './') {
 42                                 uri += strarg.substr(1);
 43                             }
 44                             else {
 45                                 uri += '/' + strarg;
 46                             }
 47                     }
 48                 }
 49             }
 50         }
 51 		return uri;
 52     }

 55      * 	@private
 56      */
 57     tent.persisters.rest.createItemMapper = function(defaultOptions){
 58     
 59         if (!defaultOptions) {
 60             defaultOptions = {};
 61         }
 62         return function(data, options, map){
 63             var items;
 64             if (data instanceof Array) {
 65                 items = data;
 66             }
 67             else {
 68                 if (options.multi) {
 69                     var multiSelector = options.multiSelector || defaultOptions.multiSelector ||
 70                     function(d, opt){
 71                         var propName = opt.multiSelectorProperty || defaultOptions.multiSelectorProperty;
 72                         if (propName) {
 73                             return d[propName];
 74                         }
 75                         else {
 76                             for (var prop in d) {
 77                                 if (d[prop] instanceof Array) {
 78                                     return d[prop];
 79                                 }
 80                             }
 81                         }
 82                     };
 83                     items = multiSelector(data, options);
 84                 }
 85                 else {
 86                     var singleSelector = options.singleSelector || defaultOptions.singleSelector ||
 87                     function(d, opt){
 88                         var propName = opt.singleSelectorProperty || defaultOptions.singleSelectorProperty;
 89                         if (propName) {
 90                             return [d[propName]];
 91                         }
 92                         else {
 93                             return [d];
 94                         }
 95                     };
 96                     items = singleSelector(data, options);
 97                 }
 98             }
 99             
100             if (items) {
101                 for (var i = 0, l = items.length; i < l; i++) {
102                     var item = items[i];
103                     if (options.multi) {
104                         var itemSelector = options.multiItemSelector || defaultOptions.multiItemSelector ||
105                         function(d, opt){
106                             var propName = opt.multiItemSelectorProperty || defaultOptions.multiItemSelectorProperty;
107                             if (propName) {
108                                 return d[propName];
109                             }
110                             else {
111                                 return d;
112                             }
113                         };
114                         item = itemSelector(item, options);
115                     }
116                     
117                     if (item) {
118                         var itemTransformer = options.itemTransformer || defaultOptions.itemTransformer;
119                         if (itemTransformer) {
120                             item = itemTransformer(item, options);
121                         }
122                     }
123                     if (item) {
124                         map(item, options);
125                     }
126                 }
127             }
128         }
129     }

134     tent.persisters.rest.createChangeSerializer = function(defaultOptions){
135     
136         if (!defaultOptions) {
137             defaultOptions = {};
138         }
139         return function(items, options){
140             var data, chg;
141             
142             var serializer = options.itemSerializer || defaultOptions.itemSerializer ||
143             function(local, op){
144                 if (local instanceof tent.entities.EntityLink) {
145                     // EntityLinks not supported
146                     return null;
147                 }
148                 var rmt;
149 				if (local.__changeState__ === tent.entities.ChangeStates.DELETED) {
150 					rmt = {};
151 					if (local.id) {
152 						rmt._id = tent.pget(local, 'id');
153 					}
154 					if (local.rev) {
155 						rmt.rev = tent.pget(local, 'rev');
156 					}
157 				}
158 				else {
159 					rmt = tent.clone(local, {
160 						deep: true,
161 						onlyOwnProperties: true,
162 						attachedObjects: false,
163 						skipPrivates: true
164 					});
165 				}
166 				if (local._id) {
167 					rmt._id = tent.pget(local, '_id');
168 				}
169 				if (local._rev) {
170 					rmt._rev = tent.pget(local, '_rev');
171 				}
172 
173                 if (local.__changeState__ === tent.entities.ChangeStates.DELETED) {
174                     rmt._deleted = true;
175                 }
176 				return rmt;
177             };
178             
179             
180             if (items instanceof Array) {
181             
182 				if (options.bulk){					
183 	                var dataItems = [];
184 	                var batch = options.batch || defaultOptions.batch || 0;
185 					var batchSize = batch;
186 					if ((!batch) || (batch < 1)){
187 						batchSize = items.length;
188 					}
189 					var limit = (options.offset || 0)+ batchSize;
190 					if (options.limit && options.limit < limit){
191 						limit = options.limit;
192 					}
193 	                for (var i = (options.offset || 0), l = limit; i < l; i++) {
194 	                    dataItems.push(serializer(items[i], options));
195 	                }
196                 
197 	                var wrapperProp = options.saveWrapperProperty || defaultOptions.saveWrapperProperty || 'docs';
198 	                var wrapper = options.itemsSaveWrapper || defaultOptions.itemsSaveWrapper ||
199 	                function(ditems, op){
200 	                    var d = {};
201 	                    d[wrapperProp] = ditems;
202 	                    return d;
203 	                };
204 	                return wrapper(dataItems, options);
205 				}else{
206                     return serializer(items[options.offset || 0], options);
207 				}
208 
209             }
210             else 
211                 if (typeof items == 'object') {
212                     return serializer(items, options);
213                 }
214         }

220     tent.persisters.rest.createChangeResponseMapper = function(defaultOptions){
221     
222         if (!defaultOptions) {
223             defaultOptions = {};
224         }
225         return function(data, options, map){
226 
227 				
228             var items;
229             if (data instanceof Array) {
230                 items = data;
231             }
232             else {
233 			
234 				var resultSelector = options.resultSelector || defaultOptions.resultSelector ||
235 				function(d, opt){
236 					if (!options.bulk){
237 						return [d];
238 					}
239 					var propName = opt.resultSelectorProperty || defaultOptions.resultSelectorProperty;
240 					if (propName) {
241 						return d[propName];
242 					}
243 					else {
244 						for (var prop in d) {
245 							if (d[prop] instanceof Array) {
246 								return d[prop];
247 							}
248 						}
249 					}
250 				};
251 				items = resultSelector(data, options);
252             }
253             
254             if (items) {
255                 for (var i = 0, l = items.length; i < l; i++) {
256                     var item = items[i];
257                     
258                     var itemSelector = options.resultItemSelector || defaultOptions.resultItemSelector ||
259                     function(d, opt){
260 						if (!opt.bulk){
261 							return d;
262 						}
263                         var propName = opt.resultItemSelectorProperty || defaultOptions.resultItemSelectorProperty;
264                         if (propName) {
265                             return d[propName];
266                         }
267                         else {
268                             return d;
269                         }
270                     };
271                     item = itemSelector(item, options);
272                     
273                     if (item) {
274                         var itemTransformer = options.resultItemTransformer || defaultOptions.resultItemTransformer;
275                         if (itemTransformer) {
276                             item = itemTransformer(item, options);
277                         }
278                     }
279                     if (item) {
280                     
281                     
282                         var localChangeFinder = options.localChangeFinder || defaultOptions.localChangeFinder ||
283                         function(ctx, remote, op){
284                             if (ctx.hasChanges()) {
285                                 for (var j = 0, l = ctx.changes.items.length; j < l; j++) {
286                                     if (typeof remote.id != 'undefined' && ctx.changes.items[j]._id === remote.id) {
287                                         return j;
288                                     }
289                                     if (typeof remote.id != 'undefined' && ctx.changes.items[j].id === remote.id) {
290                                         return j;
291                                     }
292                                     if (typeof remote._id != 'undefined' && ctx.changes.items[j]._id === remote._id) {
293                                         return j;
294                                     }
295                                     if (typeof remote._id != 'undefined' && ctx.changes.items[j].id === remote._id) {
296                                         return j;
297                                     }
298                                 }
299                             }
300                         };
301                         
302                         var localChangeIndex = localChangeFinder(options.context, item, options);
303                         
304                         if (!(typeof localChangeIndex == 'number' && localChangeIndex >= 0)) {
305 							if (!(options.unorderedResults || defaultOptions.unorderedResults)) {
306 								// if results are in order, the first change in the list, is next
307 								localChangeIndex = options.offset || 0;
308 							}
309 						}
310 						
311                         if (typeof localChangeIndex == 'number' && localChangeIndex >= 0) {
312                             map(item, localChangeIndex, options);
313                         }
314                     }
315                 }
316             }
317         }
318     }

320     /**
321      * Creates a new REST persister
322      * @class a REST persister
323      * @constructor
324      */
325     tent.persisters.rest.RestPersister = function RestPersister(){
326     
327         /**
328          * Absolute base URI for this persister
329          * @field
330          * @type String
331          */
332         this.baseUri = 'http://127.0.0.1:5984/mydb';
333         
334         /**
335          * Indicates if this persister is saving changes
336          * @field
337          * @type Boolean
338          */
339         this.saving = false;
340         
341         /**
342          * Errors ocurred while saving
343          * @field
344          * @type Array
345          */
346         this.savingErrors = [];
347         
348         /**
349          * Errors ocurred while loading
350          * @field
351          * @type Array
352          */
353         this.loadingErrors = [];
354         
355         /**
356          * function that maps items from data obtained on load
357          * @field
358          * @type function()
359          */
360         this.loadItemMapper = null;
361         
362         /**
363          * function that finds the local version of a loaded item
364          * @field
365          * @type function()
366          */
367         this.localItemFinder = function(context, remote){
368             return context.filter(function(item){
369                 if ((typeof remote._id != 'undefined' && item._id === remote._id) ||
370                 (typeof remote.id != 'undefined' && item.id === remote.id)) {
371                     return true;
372                 }
373                 return false;
374             })[0];
375         };
376         
377         /**
378          * function that compares a local item version with a loaded item
379          * @field
380          * @type function()
381          */
382         this.versionEquals = function(local, remote){
383             return local._rev === remote._rev;
384         };
385         
386         /**
387          * function that updates local version field from a save response
388          * @field
389          * @type function()
390          */
391         this.updateLocalVersion = function(local, remote){
392             local._rev = remote.rev || remote._rev;
393 			if (typeof remote.id != 'undefined' && remote.id !== local._id){
394 				local._id = remote.id;
395 			}
396         };
397         
398         /**
399          * function that indicates if a loaded item is deleted
400          * @field
401          * @type function()
402          */
403         this.isDeleted = function(remote){
404             return remote._deleted;
405         };
406         
407         /**
408          * function that serializes a change item
409          * @field
410          * @type function()
411          */
412         this.changeSerializer = null;
413         
414         /**
415          * function that maps change responses to local changes
416          * @field
417          * @type function()
418          */
419         this.changeResponseMapper = null;
420         
421         /**
422          * URI for bulk save operations (or null to use individual save requests)
423          * @field
424          * @type String
425          */
426         this.bulkSaveUri = null;
427 		
428 		/**
429 		 * Returns the URI to persist an item change
430          * @field
431          * @type function()
432 		 */
433 		this.getChangeItemUri = function(change){
434 			if (change instanceof tent.entities.EntityLink){
435 				// entity link not supported
436 				return null;
437 			}
438 			var uri = change._id || change.id || '/';
439 			if (change.__changeState__ == tent.entities.ChangeStates.DELETED){
440 				var hasQuery=false;				
441 				if (change._rev){
442 					uri+=(hasQuery ? '&':'?')+ 'rev='+change._rev;
443 					hasQuery = true;
444 				}else if (change.rev){
445 					uri+=(hasQuery ? '&':'?')+ 'rev='+change.rev;
446 					hasQuery = true;
447 				}
448 			} 
449 			return uri;
450 		}
451 				
452 		/**
453 		 * Returns the http method to persist an item change
454          * @field
455          * @type function()
456 		 */
457 		this.getChangeItemMethod = function(change){
458 			if (change.__changeState__ === tent.entities.ChangeStates.DELETED){
459 				return 'DELETE';
460 			}
461 			if (change.__changeState__ === tent.entities.ChangeStates.MODIFIED){
462 				return 'PUT';
463 			}	
464 			if (change._id || change.id){
465 				return 'PUT';
466 			}	
467 			return 'POST';
468 		}
469     }

472      * 	Load entities from an URL into a {@link tent.entities.Context}
473      *  @param {tent.entities.Context} context
474      *  @param {String} url that returns a JSON representation of entities
475      *  @param {Object} [options] loading options
476      *  @param {Boolean} [options.multi] indicates if the url will return a collection of entities
477      *  @param {String} [options.credentials] http basic auth credentials in the form 'username:password' and base64 encoded
478      *  @param {Object} [options.method] http method, default 'GET'
479      *  @param {function()} [options.complete] function to call when operation is complete
480      */
481     tent.persisters.rest.RestPersister.prototype.load = function(context, url, options){
482         try {
483             if (!context) {
484                 throw 'a context is required for loading entities';
485             }
486             if (!url) {
487                 throw 'an url must be specified to load';
488             }
489             if (this.loading) {
490                 throw 'already loading entities';
491             }
492             this.loading = true;
493             var persister = this;
494             
495             if (!options) {
496                 options = {};
497             }
498             var comp = options.complete;
499 			/**
500 			 * @ignore
501 			 */
502             options.complete = function(r){
503                 persister.loading = false;
504                 if (r.error) {
505                     persister.loadingErrors.push(r.error);
506                 }
507                 if (comp) {
508                     comp(r);
509                 }
510             };
511             
512             this.__load__(context, url, options);
513         } 
514         catch (err) {
515             this.loadingErrors.push(err);
516             this.loading = false;
517             if (options.complete) {
518                 options.complete({
519                     error: err
520                 });
521             }			
522 		}
523     }

528     tent.persisters.rest.RestPersister.prototype.__load__ = function(context, url, options){
529     
530         if (typeof jQuery == 'undefined' || typeof jQuery.ajax == 'undefined') {
531             throw 'jQuery.ajax is required in order to load entities';
532         }

534         url = tent.persisters.rest.uriCombine(this.baseUri, url);
535         
536         //options.credentials = 'YmVuamFtaW5lOnB5TGdvckNM';
537         jQuery.ajax({
538             context: {
539                 persister: this,
540                 context: context,
541                 options: options
542             },
543             url: url,
544             beforeSend: function(req){
545                 if (options.credentials) {
546                     req.setRequestHeader("Origin", document.location.protocol + "//" + document.location.host);
547                     req.setRequestHeader("Authorization", "Basic " + options.credentials);
548                 }
549             },
550             type: options.method || 'GET',
551             cache: !!options.cache,
552             dataType: 'json',
553             success: function(data, textStatus, req){
554                 var result = {
555                     persister: this,
556                     options: options
557                 };
558                 try {
559                     this.persister.__processLoadResponse__(this.context, data, options, result);
560                     result.ok = true;
561                 } 
562                 catch (err) {
563                     result.error = err;
564                 }
565                 if (options.complete) {
566                     options.complete(result);
567                 }
568             },
569             error: function(req, textStatus, error){
570                 var result = {
571                     persister: this,
572                     options: options,
573                     error: {
574                         type: textStatus,
575                         errorThrown: error
576                     }
577                 };

579                     options.complete(result);
580                 }
581             }
582         });

585     /**
586      * 	@private
587      */
588     tent.persisters.rest.RestPersister.prototype.__processLoadResponse__ = function(context, data, options, result){
589         // load entities as unchanged
590         
591         if (options.context !== context) {
592             options.context = context;
593         }
594         if (!this.loadItemMapper) {
595             this.loadItemMapper = tent.persisters.rest.createItemMapper();
596         }
597 		var persister = this;
598         this.loadItemMapper(data, options, function(doc, opt){
599         
600             var local = persister.localItemFinder(context, doc);
601             
602             if (local) {
603                 if (local.__changeState__ === tent.entities.ChangeStates.MODIFIED ||
604                 local.__changeState__ === tent.entities.ChangeStates.DELETED) {
605                     if (persister.versionEquals(local, doc)) {
606                         // revision unchanged, local version is newer
607                     }
608                     else {
609                         // revision changed, somebody else changed server version, conflict!
610                         local.__loadErrors__.push({
611                             error: 'conflict',
612                             reason: 'document modified recently by another user'
613                         });
614                     }
615                 }
616                 else {
617                     if (persister.isDeleted(doc)) {
618                         // item deleted on the server
619                         context.remove(local);
620                         context.detach(local);
621                     }
622                     else {
623                         // update local item with remote version
624                         tent.pset(local, doc, true);
625 						
626 						// track changes in complex properties
627 						if (context.changeHandler){
628 							context.changeHandler.trackComplexProperties(local);							
629 						}
630                     }
631                     if (local.__loadErrors__) {
632                         delete local.__loadErrors__;
633                     }
634                 }
635             }
636             else {
637                 // attach new item as unchanged
638                 context.attach(doc);
639             }
640         });
641     }

643 	/**
644      * Persists all changes in a {@link tent.entities.Context}
645 	 * @param {tent.entities.Context} context
646 	 * @param {Object} [options] saving options
647 	 * @param {Boolean} [options.bulk] save in bulk mode (multiple changes in one request)
648 	 * @param {Number} [options.batch] batch size in bulk mode, number of changes to save in one request
649 	 * @param {String} [options.uri] uri to use (if none, it's use item or bulk save uri)
650      * @param {String} [options.credentials] http basic auth credentials in the form 'username:password' and base64 encoded
651      * @param {function()} [options.complete] function to call when operation is complete
652      * @param {Number} [options.maxcount] maximum count of changes to save, otherwise all changes are saved
653 	 * @param {Number} [prevCount] number of saved changes on previous operation (internal use, for sequential requests)
654 	 */
655     tent.persisters.rest.RestPersister.prototype.saveChanges = function(context, options, prevCount){
656         try {
657             if (!context) {
658                 throw 'a context is required for saving changes';
659             }
660             if (!options) {
661                 options = {};
662             }
663             if (options.context !== context) {
664                 options.context = context;
665             }
666             if (context.hasChanges()) {
667                 if (this.saving) {
668                     throw 'Already saving changes';
669                 }
670                 this.saving = true;
671                 var persister = this;
672                 
673                 var comp = options.complete;
674 				
675 				/**
676 				 * @ignore
677 				 */
678                 options.complete = function(r){
679                     persister.saving = false;
680                     if (r.error) {
681                         persister.savingErrors.push(r.error);
682                     }
683                     else 
684                         if (typeof r.ok == 'undefined') {
685                             r.ok = true;
686                         }
687 					
688 					if ((r.ok && !r.error) && options.context.hasChanges() &&
689 						(!options.maxcount || (options.maxcount > r.count))) {
690 						persister.saveChanges(options.context, options, r.count);
691 					} else {						
692 	                    if (comp) {
693 	                        comp(r);
694 	                    }
695 					}
696                 };
697                 
698                 this.__persist__(context, options, prevCount);
699             }
700             else {
701                 if (options.complete) {
702                     options.complete({
703                         ok: true,
704                         nochanges: true
705                     });
706                 }
707             }
708         } 
709         catch (err) {
710             this.savingErrors.push(err);
711             this.saving = false;
712             if (options.complete) {
713                 options.complete({
714 					persister: this,
715 					options: options,
716                     error: err
717                 });
718             }
719         }
720     }

722     /**
723      * 	@private
724      */
725     tent.persisters.rest.RestPersister.prototype.__persist__ = function(context, options, prevCount){
726     
727         if (typeof jQuery == 'undefined' || typeof jQuery.ajax == 'undefined') {
728             throw 'jQuery.ajax is required in order to persist changes';
729         }
730 		
731 		var url, method, data;
732 		var change;
733         
734         if (options.bulk) {        
735             // bulk save
736 			var bulkUri = options.uri || this.bulkSaveUri;
737 			if (!bulkUri){
738 				throw 'no bulk save uri provided';
739 			}
740             url = tent.persisters.rest.uriCombine(this.baseUri, bulkUri);            
741             data = this.__getPersistData__(context, options);
742 		} else {
743 			// save an individual change
744 			change = context.changes.items[options.offset || 0];
745 			method = this.getChangeItemMethod(change);
746 			var changeUri = this.getChangeItemUri(change);
747 			if (changeUri) {
748 				url = tent.persisters.rest.uriCombine(this.baseUri, changeUri);
749 				data = this.__getPersistData__(context, options);
750 			}
751 		}		
752 				
753         if (data == null) {
754             var result = {
755                 persister: this,
756 				options: options,
757                 ok: true,
758                 nochanges: true
759             };
760             if (options.complete) {
761                 options.complete(result);
762             }
763         }
764         else {
765             jQuery.ajax({
766                 context: {
767                     persister: this,
768                     context: context,
769 					options: options,
770 					prevCount: prevCount
771                 },
772                 url: url,
773                 beforeSend: function(req){
774                     if (options.credentials) {
775                         req.setRequestHeader("Origin", document.location.protocol + "//" + document.location.host);
776                         req.setRequestHeader("Authorization", "Basic " + options.credentials);
777                     }
778                 },
779                 type: method || options.method || 'POST',
780                 cache: false,
781                 data: JSON.stringify(data),
782                 dataType: 'json',
783                 contentType: 'application/json',
784                 success: function(data, textStatus, req){
785                     var result = {
786                         persister: this.persister,
787 						options: this.options
