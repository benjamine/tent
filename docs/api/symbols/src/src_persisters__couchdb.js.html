1 /**
  2  * @requires tent.persisters.rest
  3  * @name tent.persisters.couchdb
  4  * @namespace Apache CouchDB RESTful Persister
  5  */
  6 tent.declare('tent.persisters.couchdb', function(){
  7 
  8     /**
  9      * Creates a new CouchDB RESTful persister
 10      * @class a CouchDB RESTful persister
 11      * @extends tent.persisters.rest.RestPersister
 12      * @constructor
 13      */
 14     tent.persisters.couchdb.CouchDBPersister = function CouchDBPersister(){
 15     
 16         tent.persisters.rest.RestPersister.apply(this, arguments);
 17         /**
 18          * URI for bulk save operations (or null to use individual save requests)
 19          * @field
 20          * @type String
 21          */
 22         this.bulkSaveUri = '_bulk_docs';
 23         
 24         this.loadItemMapper = tent.persisters.rest.createItemMapper({
 25             multiSelectorProperty: 'rows',
 26             multiItemSelectorProperty: 'doc'
 27         });
 28     }
 29     
 30     tent.mixin(tent.persisters.couchdb.CouchDBPersister, tent.persisters.rest.RestPersister);
 31     
 32 });
