  8 	/**
  9 	 * Adds a reverse property definition to a Javascript Object
 10 	 * @private
 11 	 * @param {Object} obj
 12 	 * @param {String} name property name in obj
 13 	 * @param {Object} [prop] reverse property definition
 14 	 * @param {String} [prop.cardinality] cardinality, one of: '11','1n','n1','nm', default is '11'
 15 	 * @param {Object} [prop.reverse] reverse property name on the referenced object(s)
 16 	 */
 17     tent.changes.reverseProperties.setReverseProperty = function(obj, name, prop){
 18 
 19         if (!prop) {
 20             prop = {};
 21         }
 22         if (!prop.cardinality) {
 23             prop.cardinality = '11';
 24         }
 25         else {
 26             prop.cardinality = prop.cardinality.toLowerCase();
 27         }
 28         if (!prop.reverse) {
 29             prop.reverse = name + 'Of';
 30         }
 31 
 32         var reverseProp = {
 33             name: prop.reverse,
 34             cardinality: prop.cardinality
 35         };
 36         if (prop.cardinality.substr(0, 1) === 'n' || prop.cardinality.substr(0, 1) === 'm') {
 37             reverseProp.isArray = true;
 38         }
 39         if (!obj.__reverseProperties__) {
 40             obj.__reverseProperties__ = {};
 41         }
 42 
 43         if (prop.cardinality.substr(1, 1) === 'n' || prop.cardinality.substr(1, 1) === 'm') {
 44 
 45             if (typeof obj[name] == 'undefined' || !(obj[name] instanceof Array)) {
 46 
 47                 obj[name] = [];
 48 
 49             }
 50 
 51             obj.__reverseProperties__[name] = reverseProp;
 52             obj[name].__parent__ = obj;
 53             obj[name].__reverseProperty__ = reverseProp;
 54 
 55         }
 56         else {
 57 
 58             if (typeof obj[name] == 'undefined') {
 59                 obj[name] = null;
 60             }
 61             obj.__reverseProperties__[name] = reverseProp;
 62         }
 63     }

 66 	 * Adds reverse property definitions to a Javascript Object
 67 	 * @private
 68 	 * @param {Object} obj
 69 	 * @param {Object} properties an object which properties are property definitions, properties with reverse=true will be synced
 70 	 */
 71     tent.changes.reverseProperties.setReverseProperties = function(obj, properties){
 72         if (properties) {
 73             for (var name in properties) {
 74                 if (properties[name].reverse) {
 75 
 76                     tent.changes.reverseProperties.setReverseProperty(obj, name, properties[name]);
 77 
 78                 }
 79             }
 80         }
 81     }

103 	/**
104 	 * Applies this Set to all arguments
105 	 * @example
106 	 *   // set reverse properties, start synchronization and logging
107 	 *   var Person = new tent.changes.reverseProperties.Set({
108 	 *   	father: { reverse: 'children', cardinality: 'n1' },
109 	 *   	children: { reverse: 'father', cardinality: '1n' },
110 	 *   	friends: { reverse: 'friends', cardinality: 'nm' },
111 	 *   });
112 	 *   Person.applyTo(p1,p2,p3).bind({ log: true });
113 	 *   p1.father = p2;
114 	 *   p2.friends.push(p3);
115 	 *   p3.children.push(p2);
116 	 *   
117 	 *   // assert reverse properties are in sync
118 	 *   strictEqual(p1, p2.children[0]);
119 	 *   strictEqual(p2, p3.friends[0]);
120 	 *   strictEqual(p3, p2.father);
121 	 *   
122 	 * @return {Array} an Array with all arguments and a 'bind' method, similar to {@link tent.changes.bind} that automatically includes reverse property sinchronization.
123 	 */
124     tent.changes.reverseProperties.Set.prototype.applyTo = function(){
125         var result = [];
126         
127         // fluent bind method (eg: MySet.applyTo(elem1,elem2).bind({log:true});)
128 		/**
129 		 * @inner
130 		 * @param {Object} options
131 		 */
132         result.bind = function(options){
133         
134             if (!options) {
135                 options = {};
136             }
137             
138             options.reverseProperties = true;
139             
140             for (var i = 0; i < this.length; i++) {
141                 var obj = this[i];
142                 if (obj.__reverseProperties__) {
143                 
144                     // bind object to reverse property handler
145                     tent.changes.bind(obj, options);
146                     
147                     // bind also Array reverse properties (eg: father.children, user.friends)
148                     var revProps = obj.__reverseProperties__;
149                     for (var propName in revProps) {
150                         if (revProps.hasOwnProperty(propName)) {
151                             if (revProps[propName].cardinality &&
152                             (revProps[propName].cardinality[1] == 'n' ||
153                             revProps[propName].cardinality[1] == 'm')) {
154                                 if (!(obj[propName] instanceof Array)) {
155                                     // something changed the Array set on applyTo, reset
156                                     obj[propName] = [];
157                                     obj[propName].__parent__ = obj;
158                                     obj[propName].__reverseProperty__ = obj.__reverseProperties__[propName];
159                                 }
160                                 tent.changes.bind(obj[propName], options);
161                             }
162                         }
163                     }
164                 }
165             }
166         }
167         
168         if (this.properties) {
169 
170             for (var i = 0; i < arguments.length; i++) {
171                 tent.changes.reverseProperties.setReverseProperties(arguments[i], this.properties);
172             }
173             result.push.apply(result, arguments);
174 
175         }
176         
177         return result;
178     }

185 	/**
186 	 * Gets the reverse property change handler.
187 	 * It keeps reverse properties synchronized.
188 	 * @return {function()} reverse property change handler
189 	 */
190     tent.changes.reverseProperties.getHandler = function(){
191         if (!reversePropertyHandler) {
192 			/**
193 			 * @inner
194 			 */
195             reversePropertyHandler = function(change){
196             
197                 // TODO: handle MANYCHANGES eventType
198                 
199                 if (change.eventType == tent.changes.EventTypes.ADDED) {
200                     var items = change.data.items;
201                     if (items && change.subject && change.subject.__reverseProperty__) {
202                         var reverseProp = change.subject.__reverseProperty__;
203                         var parent = change.subject.__parent__;
204                         for (var i = 0, l = items.length; i < l; i++) {
205                             var item = items[i];
206                             if (reverseProp.isArray) {
207                                 if (!item[reverseProp.name]) {
208                                     item[reverseProp.name] = [];
209                                 }
210                                 if (item[reverseProp.name].indexOf(parent) < 0) {
211                                     item[reverseProp.name].push(parent);
212                                 }
213                             }
214                             else {
215                                 if (item[reverseProp.name] != parent) {
216                                     if (typeof item[reverseProp.name] != 'undefined' && item[reverseProp.name] != null) {
217                                         throw new Error('this.' + reverseProp.name + ' is set. Remove this from its current array before adding it to this array');
218                                     }
219                                     item[reverseProp.name] = parent;
220                                 }
221                             }
222                         }
223                     }
224                 }
225                 else 
226                     if (change.eventType == tent.changes.EventTypes.REMOVED) {
227                         var items = change.data.items;
228                         if (items && change.subject && change.subject.__reverseProperty__) {
229                             var reverseProp = change.subject.__reverseProperty__;
230                             var parent = change.subject.__parent__;
231                             for (var i = 0, l = items.length; i < l; i++) {
232                                 var item = items[i];
233                                 if (reverseProp.isArray) {
234                                     var reverseArray = item[reverseProp.name];
235                                     if (reverseArray) {
236                                         for (var ii = 0, ll = reverseArray.length; ii < ll; ii++) {
237                                             if (reverseArray[ii] === parent) {
238                                                 reverseArray.splice(ii, 1);
239                                                 ii--;
240                                                 ll--;
241                                             }
242                                         }
243                                     }
244                                 }
245                                 else {
246                                     if (item[reverseProp.name] === parent) {
247                                         item[reverseProp.name] = null;
248                                     }
249                                 }
250                             }
251                         }
252                     }
253                     else 
254                         if (change.eventType == tent.changes.EventTypes.CHANGED) {
255                             var items = change.data.items;
256 
257                             if (change.data.propertyName && change.subject && change.subject.__reverseProperties__) {
258                                 var reverseProp = change.subject.__reverseProperties__[change.data.propertyName];
259 
260                                 if (reverseProp) {
261                                     var item;
262 
263                                     if (reverseProp.isArray) {
264                                         if (item = change.data.oldValue) {
265                                             var reverseArray = item[reverseProp.name];
266                                             if (reverseArray instanceof Array) {
267                                                 for (var i = 0, l = reverseArray.length; i < l; i++) {
268                                                     if (reverseArray[i] === change.subject) {
269                                                         reverseArray.splice(i, 1);
270                                                         i--;
271                                                         l--;
272                                                     }
273                                                 }
274                                             }
275                                         }
276                                         if (item = change.data.current) {
277                                             if (!item[reverseProp.name]) {
278                                                 item[reverseProp.name] = [];
279                                             }
280                                             if (item[reverseProp.name] instanceof Array) {
281                                                 if (item[reverseProp.name].indexOf(change.subject) < 0) {
282                                                     item[reverseProp.name].push(change.subject);
283                                                 }
284                                             }
285                                         }
286                                     }
287                                     else {
288 
289                                         if (item = change.data.oldValue) {
290                                             if (item[reverseProp.name] == change.subject) {
291                                                 item[reverseProp.name] = null;
292                                             }
293                                         }
294 
295                                         if (item = change.data.current) {
296                                             if (item[reverseProp.name] != change.subject) {
297 
298                                                 item[reverseProp.name] = change.subject;
299                                             }
300                                         }
301                                     }
302                                 }
303                             }
304                         }
305             }
306         }
307         return reversePropertyHandler;
308     }
