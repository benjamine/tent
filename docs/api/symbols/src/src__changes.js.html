 30     /**
 31      * @return {Number} the best property setter intercept mode support in the current environment, a value of {@link tent.changes.PropertyInterceptModes}
 32      */
 33     tent.changes.getPropertyInterceptMode = function(){
 34         if (typeof _PropertyInterceptMode == 'undefined') {
 35             // detect available property intercept mode
 36             
 37             var anobject = {};
 38             if (Object.defineProperty) {
 39                 try {
 40                     Object.defineProperty(anobject, 'myproperty', {
 41                         get: function(){
 42                             return this._myproperty;
 43                         },
 44                         set: function(value){
 45                             this._myproperty = value;
 46                         }
 47                     })
 48                     anobject.myproperty = 'my value';
 49                     if (anobject.myproperty === 'my value' && anobject._myproperty === 'my value') {
 50                         _PropertyInterceptMode = tent.changes.PropertyInterceptModes.DEFINEPROPERTY;
 51                     }
 52                 } 
 53                 catch (err) {
 54                     // defineProperty on javascript objects not supported test DOM object
 55                     if (document && document.createElement) {
 56                         var adomobject = document.createElement('span');
 57                         try {
 58                             Object.defineProperty(adomobject, 'myproperty', {
 59                                 get: function(){
 60                                     return this._myproperty;
 61                                 },
 62                                 set: function(value){
 63                                     this._myproperty = value;
 64                                 }
 65                             })
 66                             adomobject.myproperty = 'my value';
 67                             if (adomobject.myproperty === 'my value' && adomobject._myproperty === 'my value') {
 68                                 _PropertyInterceptMode = tent.changes.PropertyInterceptModes.DEFINEPROPERTYONLYDOM;
 69                             }
 70                         } 
 71                         catch (err) {
 72                             // defineProperty on dom objects not supported either
 73                         }
 74                     }
 75                 }
 76             }
 77             if (typeof _PropertyInterceptMode == 'undefined') {
 78                 anobject = {};
 79                 if (anobject.__defineSetter__) {
 80                     try {
 81                         anobject.__defineGetter__('myproperty', function(){
 82                             return this._myproperty;
 83                         });
 84                         anobject.__defineSetter__('myproperty', function(value){
 85                             this._myproperty = value;
 86                         });
 87                         anobject.myproperty = 'my value';
 88                         if (anobject.myproperty === 'my value' && anobject._myproperty === 'my value') {
 89                             _PropertyInterceptMode = tent.changes.PropertyInterceptModes.DEFINESETTER;
 90                         }
 91                     } 
 92                     catch (err) {
 93                     }
 94                 }
 95             }
 96             if (typeof _PropertyInterceptMode == 'undefined') {
 97                 _PropertyInterceptMode = tent.changes.PropertyInterceptModes.NONE;
 98             }
 99         }
100         return _PropertyInterceptMode;
101     }

138 	 * Functions that stringify a {link tent.changes.Change} indexed by eventType (a value of {@link tent.changes.EventTypes})
139 	 * @type function()[]
140 	 */
141     tent.changes.ChangeStringifiers = [];
142     
143     tent.changes.Change.prototype.toString = function(){
144     
145     
146         if (typeof tent.changes.ChangeStringifiers[this.eventType] != 'undefined') {
147         
148             tent.changes.ChangeStringifiers[this.eventType](this);
149             
150         }
151         else {
152         
153             // default change stringifier
154             
155             var change = this;
156             var message = tent.changes.EventTypes.getName(change.eventType);
157             if (change.eventType == tent.changes.EventTypes.MANYCHANGES) {
158                 message += ': ';
159                 if (change.data.length <= 3) {
160                     for (var i = 0, l = change.data.length; i < l; i++) {
161                         message += change.data[i].toString() + ",";
162                     }
163                 }
164                 else {
165                     // digest by event types
166                     var et = {};
167                     for (var i = 0, l = change.data.length; i < l; i++) {
168                         et[change.data[i].eventType] = (et[change.data[i].eventType] || 0) + 1;
169                     }
170                     for (var e in et) {
171                         message += tent.changes.EventTypes.getName(e) + '(' + et[e] + '),';
172                     }
173                 }
174             }
175             else {
176                 if (change.data.propertyName) {
177                     message += ' ' + change.data.propertyName;
178                     if (change.eventType == tent.changes.EventTypes.CHANGING) {
179                         message += ' from ' + change.data.current + ' to ' + change.data.newValue;
180                     }
181                     else 
182                         if (change.eventType == tent.changes.EventTypes.CHANGED) {
183                             message += ' from ' + change.data.oldValue + ' to ' + change.data.current;
184                         }
185                     message += ' (on ' + change.subject + ')';
186                 }
187                 if (change.data.items) {
188                     var items = change.data.items;
189                     message += ' ' + items.length + (items.length > 1 ? ' items' : ' item');
190                     if (typeof change.data.index != 'undefined') {
191                         message += ' at index ' + change.data.index;
192                     }
193                     if (change.data.propertyName) {
194                         message += ' (' + change.subject.__parent__ + '.' + change.data.propertyName + ')';
195                     }
196                     else {
197                         message += ' ([' + items + '])';
198                     }
199                 }
200             }
201             return '[Change: ' + message + ']';
202         }
203     }

268 	/**
269 	 * Sets an object property value. If subject is an observed object, it ensures that its {@link tent.changes.Observable} is notified.
270 	 * 
271 	 * Its useful when no property intercept mode is supported {link tent.changes.PropertyInterceptModes}.
272 	 * 
273 	 * @example
274 	 * 
275 	 * 		tent.pset(subject, propertyName, value); // is equivalent to...
276 	 * 		subject[propertyName] = value; 
277 	 * 
278 	 * 		tent.pset(subject, source); // is equivalent to...
279 	 * 		subject[property1] = source[property1]; 
280 	 * 		subject[property2] = source[property2];
281 	 * 		// etc. (every property in source is copied)
282 	 * 
283 	 * @param {Object} subject the modified object
284 	 * @param {String|Object} propertyName the property to set, or an object to copy properties from
285 	 * @param [value] the value to set on property
286 	 * @param {Boolean} [skipInterceptors] if true values are set without calling interceptors
287 	 * @return the set value
288 	 */
289     tent.pset = function(subject, propertyName, value, skipInterceptors){
290 		
291 		if (typeof propertyName == 'object') {
292 			for (var prop in propertyName) {
293 				if (propertyName.hasOwnProperty(prop)) {
294 					tent.pset(subject, prop, propertyName[prop], value);
295 				}
296 			}
297 			return subject;
298 		}
299 		else {
300 		
301 			if (subject.__observable__ && subject.__observable__.interceptors &&
302 			subject.__observable__.interceptors[propertyName]) {
303 				if (skipInterceptors) {
304 					var storeProp = subject.__observable__.interceptors[propertyName]._name || propertyName;
305 					subject[storeProp] = value;
306 				}
307 				else {
308 					subject.__observable__.interceptors[propertyName].newsetter.call(subject, value);
309 				}
310 			}
311 			else {
312 				subject[propertyName] = value;
313 			}
314 			return value;
315 		}
316     }

320 	 * 
321 	 * Its useful when no property intercept mode is supported {link tent.changes.PropertyInterceptModes}.  
322 	 * Equivalent to:
323 	 * 		subject[propertyName] 
324 	 * 
325 	 * @param {Object} subject the object to read
326 	 * @param {String} propertyName the property to read
327 	 * @return the property value
328 	 */
329     tent.pget = function(subject, propertyName){
330         if (subject.__observable__ && subject.__observable__.interceptors &&
331         subject.__observable__.interceptors[propertyName]) {
332             return subject.__observable__.interceptors[propertyName].newgetter.call(subject);
333         }
334         else {
335             return subject[propertyName];
336         }
337     }

340 	 * Intercepts a property on {@link #subject}
341 	 * @param {String} name property name
342 	 * @param {String} _name back storage property name
343 	 * @param {function()} getter getter function
344 	 * @param {function()} setter setter function
345 	 * @return {Object} the new interceptor, or the already existing if there was one.
346 	 */
347     tent.changes.Observable.prototype.interceptProperty = function(name, _name, getter, setter){
348     
349         if (this.interceptors[name]) {
350             return this.interceptors[name];
351         }
352         
353         var intercept = {
354             name: name,
355             type: tent.changes.InterceptorTypes.PROPERTY,
356             newsetter: setter,
357             newgetter: getter
358         };
359         
360         if (_name) {
361             _name = defaultBackPropertyPrefix + name;
362             try {
363                 this.subject[_name] = this.subject[name];
364             } 
365             catch (error) {
366                 // error on property get, initialize as null
367                 this.subject[_name] = null;
368             }
369             intercept._name = _name;
370         }
371         
372         var mode = tent.changes.getPropertyInterceptMode();
373         
374         if ((mode == tent.changes.PropertyInterceptModes.DEFINEPROPERTY) ||
375         ((mode == tent.changes.PropertyInterceptModes.DEFINEPROPERTYONLYDOM) &&
376         (tent.isDOMObject(this.subject)))) {
377         
378             try {
379                 intercept.descriptor = Object.getOwnPropertyDescriptor(this.subject, name);
380             } 
381             catch (err) {
382                 // no descriptor found
383             }
384             var newDescriptor = { //   configurable: true,
385                 //   writable: true
386             }
387             
388             if (getter) {
389                 newDescriptor.get = getter;
390             }
391             if (setter) {
392                 newDescriptor.set = setter;
393             }
394             
395             try {
396                 Object.defineProperty(this.subject, name, newDescriptor);
397             } 
398             catch (error) {
399                 tent.log.warn('Object.defineProperty for property \'' + name + '\' failed: ' + error);
400             }
401         }
402         else 
403             if (mode == tent.changes.PropertyInterceptModes.DEFINESETTER) {
404             
405                 try {
406                     intercept.getter = this.subject.__lookupGetter__(name);
407                     intercept.setter = this.subject.__lookupSetter__(name);
408                 } 
409                 catch (err) {
410                 // no getter/setter found
411                 }
412                 if (getter) {
413                     try {
414                         this.subject.__defineGetter__(name, getter);
415                     } 
416                     catch (error) {
417                         tent.log.warn('Object.__defineGetter__ for property \'' + name + '\' failed: ' + error);
418                     }
419                 }
420                 if (setter) {
421                     try {
422                         this.subject.__defineSetter__(name, setter);
423                     } 
424                     catch (error) {
425                         tent.log.warn('Object.__defineSetter__ for property \'' + name + '\' failed: ' + error);
426                     }
427                 }
428             }
429             else {
430             // no property intercept supported
431             // use: tent.pset(this.subject,'propertyName',value);  
432             //      tent.pget(this.subject,'propertyName');  
433             }
434         
435         if (intercept) {
436             this.interceptors[name] = intercept;
437         }
438 		return intercept;
439     }

484 	/**
485 	 * Creates a setter for a property that notifies the {@link tent.changes.Observable}
486 	 * @private
487 	 * @param {String} propName
488 	 * @param {String} _propName back storage property
489 	 * @return {function()} setter function
490 	 */
491     var buildPropertySetter = function(propName, _propName){
492         return function(value){
493         
494             var current = this[_propName];
495             if (current != value) {
496             
497                 this.__observable__.notifyChange(tent.changes.EventTypes.CHANGING, {
498                     propertyName: propName,
499                     current: current,
500                     newValue: value
501                 });
502 				
503 				var notifyParent = this.__observable__.shouldNotifyParent();
504 
505 				if (notifyParent) {
506 					// notify parent object (change tracking on complex properties)
507 	                this.__observable__.notifyParentChanging({
508 						propertyName: propName,
509 	                    current: current,
510 	                    newValue: value
511 					});
512 				}
513                 
514                 this[_propName] = value;
515                 
516                 this.__observable__.notifyChange(tent.changes.EventTypes.CHANGED, {
517                     propertyName: propName,
518                     current: value,
519                     oldValue: current
520                 });
521 				
522 				if (notifyParent) {
523 					// notify parent object (change tracking on complex properties)
524 	                this.__observable__.notifyParentChanged({
525 						propertyName: propName,
526 	                    current: value,
527 	                    oldValue: current
528 					});
529 				}
530             }
531         }
532     };

556 	/**
557 	 * Intercepts properties of {@link #subject}, if new properties are added this function can be called again (it's idempotent)
558 	 * @param {Object} [options] intercepting options
559 	 * @param {function()} [options.propertyFilter] property filter, by default all properties starting with underscore are filtered
560 	 * @param {String} [options.backPropertyPrefix] prefix for back storage properties, by default is underscore
561 	 * @param {Boolean} [options.trackDomProperties] if true dom properties are intercepted too (by default false), see {@link tent.isDomObject} and {@link tent.isDomProperty}
562 	 * @return {tent.changes.Observable} this
563 	 */
564     tent.changes.Observable.prototype.interceptProperties = function(options){
565     
566         if (!options) {
567             options = {};
568         }
569         var filter = options.propertyFilter || defaultPropertyInterceptFilter;
570         var backPropertyPrefix = options.backPropertyPrefix || defaultBackPropertyPrefix;
571         
572         var filterDomProps = (!options.trackDomProperties) && this.isDOMObject();
573         
574         
575         for (var propName in this.subject) {
576         
577             if (tent.isDOMProperty(propName)) {
578                 continue;
579             }
580             
581             var valueType;
582             try {
583                 if (((valueType = typeof this.subject[propName]) != 'function') && filter(this.subject, propName)) {
584                     var _propName = backPropertyPrefix + propName;
585                     
586                     this.interceptProperty(propName, _propName, buildPropertyGetter(propName, _propName), buildPropertySetter(propName, _propName));
587                 }
588             } 
589             catch (error) {
590                 tent.log.warn('Error intercepting property \'' + propName + '\': ' + error);
591             }
592         }
593 		return this;        
594     }

608 	 * <p>

610 	 * <ul>
611 	 *   <li>setting by index, ie: array[index]=value, use array.set(index,value) instead.</li>
612 	 *   <li>setting array length property, ie: array.length = newlength, use array.setLength(newLength) instead.</li>

617 	 */
618     tent.changes.Observable.prototype.interceptArrayModifiers = function(options){

621             return this;
622         }
623

626 	   	/**
627 	   	 * @inner

629 	   	 */

631 			// length setter that uses splice to trim arrays
632             if (this.length > l) {
633                 this.splice(l, this.length - l);
634             }
635             else {
636                 this.lengh = l;
637             }
638         }
639         
640         this.interceptFunction('push', function(){
641             var index = this.length;
642             var itemsToAdd = itemsToAdd = Array.prototype.slice.call(arguments);
643             this.__observable__.notifyChange(tent.changes.EventTypes.ADDING, {
644                 items: itemsToAdd,

646                 propertyName: this.__propertyName__
647             });
648

650

660 				});
661 			}

667                 propertyName: this.__propertyName__
668             });
669 			
670 			if (notifyParent) {
671 				// notify parent object (change tracking on complex properties)

673 					type: tent.changes.EventTypes.ADDED,
674                     data: {
675 		                items: itemsToAdd,
676 		                index: index,
677 		                propertyName: this.__propertyName__
678 		            }
679 				});
680 			}
681         });
682         
683         this.interceptFunction('unshift', function(){
684             var itemsToAdd = itemsToAdd = Array.prototype.slice.call(arguments);
685             this.__observable__.notifyChange(tent.changes.EventTypes.ADDING, {
686                 items: itemsToAdd,
687                 index: 0,
688                 propertyName: this.__propertyName__
689             });
690 									
691 			var notifyParent = this.__observable__.shouldNotifyParent();
692 
693 			if (notifyParent) {
694 				// notify parent object (change tracking on complex properties)
695                 this.__observable__.notifyParentChanging({
696 					type: tent.changes.EventTypes.ADDING,
697                     data: {
698 		                items: itemsToAdd,
699 		                index: 0,
700 		                propertyName: this.__propertyName__
701 		            }
702 				});
703 			}
704 			
705             this._unshift.apply(this, arguments);
706             this.__observable__.notifyChange(tent.changes.EventTypes.ADDED, {
707                 items: itemsToAdd,
708                 index: 0,
709                 propertyName: this.__propertyName__
710             });
711 			
712 			if (notifyParent) {
713 				// notify parent object (change tracking on complex properties)

935 	 * Removes an interceptor by name, restoring the property or function original state
936 	 * @param {String} name a function or property name on {@link #subject}
937 	 */
938     tent.changes.Observable.prototype.removeInterceptor = function(name){
939         var interceptor = this.interceptors[name];
940         
941         if (interceptor._name) {
942             if (interceptor.type != tent.changes.InterceptorTypes.FUNCTION) {
943                 try {
944                     delete this.subject[interceptor.name];
945                 } 
946                 catch (error) {
947                     tent.log.warn('Error deleting property interceptor \'' + interceptor.name + '\': ' + error);
948                 }
949                 
950                 var mode = tent.changes.getPropertyInterceptMode();
951                 if (mode == tent.changes.PropertyInterceptModes.DEFINEPROPERTY ||
952                 (mode == tent.changes.PropertyInterceptModes.DEFINEPROPERTYONLYDOM &&
953                 tent.isDOMObject(this.subject))) {
954                     if (interceptor.descriptor && interceptor.descriptor.name) {
955                         Object.defineProperty(this.subject, name, interceptor.descriptor);
956                     }
957                 }
958                 else 
959                     if (mode == tent.changes.PropertyInterceptModes.DEFINESETTER) {
960                         if (interceptor.getter) {
961                             this.subject.__defineGetter__(name, interceptor.getter);
962                         }
963                         if (interceptor.setter) {
964                             this.subject.__defineSetter__(name, interceptor.setter);
965                         }
966                     }
967                     else {
968                     // no property interceptors supported, nothing to remove
969                     }
970                 
971             }
972             
973             var domObject;
974             try {
975                 this.subject[interceptor.name] = this.subject[interceptor._name];
976             } 
977             catch (error) {
978                 domObject = tent.isDOMObject(this.subject);
979                 // error setting property values are common on DOM objects
980                 if (!domObject) {
981                     tent.log.warn('Error restoring property \'' + interceptor.name + '\' value: ' + error);
982                 }
983             }
984             try {
985                 delete this.subject[interceptor._name];
986             } 
987             catch (error) {
988                 tent.log.warn('Error deleting back storage property \'' + interceptor._name + '\': ' + error);
989             }
990         }
991         else {
992             if (interceptor.type != tent.changes.InterceptorTypes.FUNCTION) {
993                 var val = this.subject[interceptor.name];
994                 delete this.subject[interceptor.name];
995                 this.subject[interceptor.name] = val;
996             }
997         }
998         delete interceptor[name];
999     }

1026 	 * If few changes have ocurred while suspended (5 or less) they are all notified in order.

1030         this.suspended = false;
1031         if (this.changesWhileSuspended) {
1032             // notify changes while suspended
1033             if (this.changesWhileSuspended.length < 5) {
1034                 // if there were few changes, notify them now

1068 	 * @param {Number} [eventType] a value of {@link tent.changes.EventTypes}, if none is provided all events are included
1069 	 * @param {Object|function()} handler a valid change handler, see {@link #isValidHandler}
1070 	 * @return {Boolean} false if no action has been performed (ie: the binding already exists)
1071 	 */
1072     tent.changes.Observable.prototype.bind = function(eventType, handler){
1073         if (!handler) {
1074             if (this.isValidHandler(eventType)) {
1075                 handler = eventType;
1076                 eventType = tent.changes.EventTypes.ANY;
1077             }
1078             else {
1079                 throw 'must specify a handler: function (Change) or and object with a handle(Change) function';
1080             }
1081         }
1082         
1083         if (!this.isValidHandler(handler)) {
1084             throw 'must specify a handler: function (Change) or and object with a handle(Change) function';
1085         }
1086         
1087         if (!eventType) {
1088             eventType = tent.changes.EventTypes.ANY;
1089         }

1091         if (!ehandlers) {
1092             ehandlers = this.handlers[eventType] = [];
1093         }
1094         for (var i = 0; i < ehandlers.length; i++) {
1095             if (ehandlers[i] == handler) {
1096                 return false;
1097             }
1098         }
1099         this.handlers[eventType].push(handler);
1100         return true;
1101     }

1380 	 */

1384             options = {};
1385         }
1386         
1387         if (typeof obj != 'object') {
1388             if (!options.deepStack) {
1389                 throw ('Cannot track changes of ' + typeof obj);
1390             }
1391             return false;
1392         }
1393         else 
1394             if (obj == null) {

1403 			return false;

1472                             changed = true;
1473                         }
1474                     }
1475                 }
1476             }
1477         }
1478         // bind special event handlers
1479         if (options.log) {
1480             // log changes
1481             if (obj.__observable__ && obj.__observable__.log(!options.remove)) {

1483             }
1484         }

1486             // keep reverse properties in sync
1487             if (options.remove) {
1488                 if (obj.__observable__ && obj.__observable__.unbind(tent.changes.reverseProperties.getHandler())) {
1489                     changed = true;
1490                 }
1491             }
